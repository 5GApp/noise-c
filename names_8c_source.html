122 int noise_name_to_id(int category, const char *name, size_t name_len)
123 {
124  const NoiseIdMapping *mapping = algorithm_names;
125  int mask = category ? NOISE_ID(0xFF, 0) : 0;
126  if (!name)
127  return 0;
128  while (mapping->name_len) {
129  if ((mapping->id & mask) == category) {
130  if (mapping->name_len == name_len &&
131  !memcmp(mapping->name, name, name_len)) {
132  return mapping->id;
133  }
134  }
135  ++mapping;
136  }
137  return 0;
138 }

158 const char *noise_id_to_name(int category, int id)
159 {
160  const NoiseIdMapping *mapping = algorithm_names;
161  int mask = category ? NOISE_ID(0xFF, 0) : 0;
162  if (id <= 0)
163  return 0;
164  while (mapping->name_len) {
165  if ((mapping->id & mask) == category) {
166  if (mapping->id == id)
167  return mapping->name;
168  }
169  ++mapping;
170  }
171  return 0;
172 }

190 static int noise_protocol_parse_field
191  (int category, const char *name, size_t len, size_t *posn,
192  int is_last, int *ok)
193 {
194  size_t start, field_len;
195  int id;
196 
197  /* If the parse already failed, then nothing further to do */
198  if (!(*ok))
199  return 0;
200 
201  /* Find the start and end of the current field */
202  start = *posn;
203  while (*posn < len && name[*posn] != '_')
204  ++(*posn);
205  field_len = *posn - start;
206 
207  /* If this is the last field, we should be at the end
208  of the string. Otherwise there should be a '_' here */
209  if (is_last) {
210  if (*posn < len) {
211  *ok = 0;
212  return 0;
213  }
214  } else {
215  if (*posn >= len) {
216  *ok = 0;
217  return 0;
218  }
219  ++(*posn); /* Skip the '_' */
220  }
221 
222  /* Look up the name in the current category */
223  id = noise_name_to_id(category, name + start, field_len);
224  if (!id)
225  *ok = 0;
226  return id;
227 }

243 int noise_protocol_name_to_id
244  (NoiseProtocolId *id, const char *name, size_t name_len)
245 {
246  size_t posn;
247  int ok;
248 
249  /* Bail out if the parameters are incorrect */
250  if (!id || !name)
251  return NOISE_ERROR_INVALID_PARAM;
252 
253  /* Parse underscore-separated fields from the name */
254  posn = 0;
255  ok = 1;
256  id->prefix_id = noise_protocol_parse_field
257  (NOISE_PREFIX_CATEGORY, name, name_len, &posn, 0, &ok);
258  id->pattern_id = noise_protocol_parse_field
259  (NOISE_PATTERN_CATEGORY, name, name_len, &posn, 0, &ok);
260  id->dh_id = noise_protocol_parse_field
261  (NOISE_DH_CATEGORY, name, name_len, &posn, 0, &ok);
262  id->cipher_id = noise_protocol_parse_field
263  (NOISE_CIPHER_CATEGORY, name, name_len, &posn, 0, &ok);
264  id->hash_id = noise_protocol_parse_field
265  (NOISE_HASH_CATEGORY, name, name_len, &posn, 1, &ok);
266  id->reserved_id = 0;
267 
268  /* If there was a parse error, then clear everything */
269  if (!ok) {
270  id->prefix_id = NOISE_PREFIX_NONE;
271  id->pattern_id = NOISE_PATTERN_NONE;
272  id->dh_id = NOISE_DH_NONE;
273  id->cipher_id = NOISE_CIPHER_NONE;
274  id->hash_id = NOISE_HASH_NONE;
275  id->reserved_id = 0;
276  return NOISE_ERROR_UNKNOWN_NAME;
277  }
278 
279  /* The name has been parsed */
280  return NOISE_ERROR_NONE;
281 }

296 static void noise_protocol_format_field
297  (int category, int id, char *name, size_t len, size_t *posn,
298  int is_last, int *err)
299 {
300  const char *alg_name;
301  size_t alg_len;
302 
303  /* If the formatting already failed, then bail out now */
304  if (*err != NOISE_ERROR_NONE)
305  return;
306 
307  /* Look up the name for the algorithm identifier */
308  alg_name = noise_id_to_name(category, id);
309  if (!alg_name) {
310  *err = NOISE_ERROR_UNKNOWN_ID;
311  return;
312  }
313  alg_len = strlen(alg_name);
314 
315  /* Will the name fit into the buffer, followed by either '_' or '\0'? */
316  if (alg_len >= (len - *posn)) {
317  *err = NOISE_ERROR_INVALID_LENGTH;
318  return;
319  }
320  memcpy(name + *posn, alg_name, alg_len);
321  *posn += alg_len;
322 
323  /* Add either a separator or a terminator */
324  if (!is_last)
325  name[(*posn)++] = '_';
326  else
327  name[*posn] = '\0';
328 }

353 int noise_protocol_id_to_name
354  (char *name, size_t name_len, const NoiseProtocolId *id)
355 {
356  size_t posn;
357  int err;
358 
359  /* Bail out if the parameters are incorrect */
360  if (!id) {
361  if (name && name_len)
362  *name = '\0'; /* Just to be safe */
363  return NOISE_ERROR_INVALID_PARAM;
364  }
365  if (!name)
366  return NOISE_ERROR_INVALID_PARAM;
367  if (!name_len)
368  return NOISE_ERROR_INVALID_LENGTH;
369 
370  /* Format the fields into the return buffer */
371  posn = 0;
372  err = NOISE_ERROR_NONE;
373  noise_protocol_format_field
374  (NOISE_PREFIX_CATEGORY, id->prefix_id, name, name_len, &posn, 0, &err);
375  noise_protocol_format_field
376  (NOISE_PATTERN_CATEGORY, id->pattern_id, name, name_len, &posn, 0, &err);
377  noise_protocol_format_field
378  (NOISE_DH_CATEGORY, id->dh_id, name, name_len, &posn, 0, &err);
379  noise_protocol_format_field
380  (NOISE_CIPHER_CATEGORY, id->cipher_id, name, name_len, &posn, 0, &err);
381  noise_protocol_format_field
382  (NOISE_HASH_CATEGORY, id->hash_id, name, name_len, &posn, 1, &err);
383 
384  /* The reserved identifier must be zero. We don't know how to
385  format reserved identifiers other than zero */
386  if (id->reserved_id != 0 && err == NOISE_ERROR_NONE)
387  err = NOISE_ERROR_UNKNOWN_ID;
388 
389  /* If an error occurred, then clear the buffer just to be safe */
390  if (err != NOISE_ERROR_NONE)
391  *name = '\0';
392 
393  /* Done */
394  return err;
395 }
