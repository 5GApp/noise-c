195 int noise_format_fingerprint
196  (int fingerprint_type, char *buffer, size_t len,
197  const uint8_t *public_key, size_t public_key_len)
198 {
199  static char const hexchars[] = "0123456789abcdef";
200  sha256_context_t sha256;
201  uint8_t hash[32];
202  size_t f_len;
203  size_t posn;
204 
205  /* Validate the parameters */
206  if (!buffer)
207  return NOISE_ERROR_INVALID_PARAM;
208  if (!len)
209  return NOISE_ERROR_INVALID_LENGTH;
210  *buffer = '\0'; /* In case we bail out with an error later */
211  if (!public_key)
212  return NOISE_ERROR_INVALID_PARAM;
213 
214  /* Validate the fingerprint type and get the desired length */
215  if (fingerprint_type == NOISE_FINGERPRINT_BASIC)
216  f_len = 16;
217  else if (fingerprint_type == NOISE_FINGERPRINT_FULL)
218  f_len = 32;
219  else
220  return NOISE_ERROR_INVALID_PARAM;
221 
222  /* Check the length of the buffer */
223  if ((f_len * 3) > len)
224  return NOISE_ERROR_INVALID_LENGTH;
225 
226  /* Hash the public key with SHA256 */
227  sha256_reset(&sha256);
228  sha256_update(&sha256, public_key, public_key_len);
229  sha256_finish(&sha256, hash);
230  noise_clean(&sha256, sizeof(sha256));
231 
232  /* Format the fingerprint in hexadecimal within the buffer */
233  for (posn = 0; posn < f_len; ++posn) {
234  uint8_t byte = hash[posn];
235  buffer[posn * 3] = hexchars[(byte >> 4) & 0x0F];
236  buffer[posn * 3 + 1] = hexchars[byte & 0x0F];
237  buffer[posn * 3 + 2] = ':';
238  }
239  buffer[f_len * 3 - 1] = '\0';
240  noise_clean(hash, sizeof(hash));
241  return NOISE_ERROR_NONE;
242 }
