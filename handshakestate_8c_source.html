65 static int noise_handshakestate_new
66  (NoiseHandshakeState **state, NoiseSymmetricState *symmetric, int role)
67 {
68  const uint8_t *pattern;
69  int requirements;
70  int dh_id;
71  uint8_t flags;
72  int err;
73 
74  /* Locate the information for the current handshake pattern */
75  pattern = noise_pattern_lookup(symmetric->id.pattern_id);
76  if (!pattern) {
77  noise_symmetricstate_free(symmetric);
78  return NOISE_ERROR_UNKNOWN_ID;
79  }
80  flags = pattern[0];
81  if (role == NOISE_ROLE_RESPONDER) {
82  /* Reverse the pattern flags so that the responder is "local" */
83  flags = noise_pattern_reverse_flags(flags);
84  }
85 
86  /* Create the HandshakeState object */
87  *state = noise_new(NoiseHandshakeState);
88  if (!(*state)) {
89  noise_symmetricstate_free(symmetric);
90  return NOISE_ERROR_NO_MEMORY;
91  }
92 
93  /* What keys do we require to be able to start the protocol? */
94  requirements = 0;
95  if (flags & NOISE_PAT_FLAG_LOCAL_STATIC) {
96  requirements |= NOISE_REQ_LOCAL_REQUIRED;
97  } if (flags & NOISE_PAT_FLAG_LOCAL_REQUIRED) {
98  requirements |= NOISE_REQ_LOCAL_REQUIRED;
99  requirements |= NOISE_REQ_LOCAL_PREMSG;
100  }
101  if (flags & NOISE_PAT_FLAG_REMOTE_REQUIRED) {
102  requirements |= NOISE_REQ_REMOTE_REQUIRED;
103  requirements |= NOISE_REQ_REMOTE_PREMSG;
104  }
105  if (symmetric->id.prefix_id == NOISE_PREFIX_PSK) {
106  requirements |= NOISE_REQ_PSK;
107  }
108  requirements |= NOISE_REQ_PROLOGUE;
109 
110  /* Initialize the HandshakeState */
111  (*state)->requirements = requirements;
112  (*state)->action = NOISE_ACTION_NONE;
113  (*state)->tokens = pattern + 1;
114  (*state)->role = role;
115  (*state)->symmetric = symmetric;
116 
117  /* Create DHState objects for all of the keys we will need later */
118  err = NOISE_ERROR_NONE;
119  dh_id = symmetric->id.dh_id;
120  if ((flags & NOISE_PAT_FLAG_LOCAL_STATIC) != 0)
121  err = noise_dhstate_new_by_id(&((*state)->dh_local_static), dh_id);
122  if ((flags & NOISE_PAT_FLAG_LOCAL_EMPEMERAL) != 0 && err == NOISE_ERROR_NONE)
123  err = noise_dhstate_new_by_id(&((*state)->dh_local_ephemeral), dh_id);
124  if ((flags & NOISE_PAT_FLAG_REMOTE_STATIC) != 0 && err == NOISE_ERROR_NONE)
125  err = noise_dhstate_new_by_id(&((*state)->dh_remote_static), dh_id);
126  if ((flags & NOISE_PAT_FLAG_REMOTE_EMPEMERAL) != 0 && err == NOISE_ERROR_NONE)
127  err = noise_dhstate_new_by_id(&((*state)->dh_remote_ephemeral), dh_id);
128 
129  /* Bail out if we had an error trying to create the DHState objects */
130  if (err != NOISE_ERROR_NONE) {
131  noise_handshakestate_free(*state);
132  return err;
133  }
134 
135  /* Ready to go */
136  return NOISE_ERROR_NONE;
137 }

436 int noise_handshakestate_set_pre_shared_key
437  (NoiseHandshakeState *state, const uint8_t *key, size_t key_len)
438 {
439  uint8_t temp[NOISE_MAX_HASHLEN];
440  NoiseHashState *hash;
441 
442  /* Validate the parameters and state */
443  if (!state || !key)
444  return NOISE_ERROR_INVALID_PARAM;
445  if (state->symmetric->id.prefix_id != NOISE_PREFIX_PSK)
446  return NOISE_ERROR_NOT_APPLICABLE;
447  if (state->action != NOISE_ACTION_NONE)
448  return NOISE_ERROR_INVALID_STATE;
449  if (!(state->requirements & NOISE_REQ_PSK))
450  return NOISE_ERROR_INVALID_STATE;
451 
452  /* If we haven't hashed the prologue yet, hash an empty one now */
453  if (state->requirements & NOISE_REQ_PROLOGUE)
454  noise_handshakestate_set_prologue(state, "", 0);
455 
456  /* Mix the pre shared key into the chaining key and handshake hash */
457  hash = state->symmetric->hash;
458  noise_hashstate_hkdf
459  (hash, state->symmetric->ck, hash->hash_len, key, key_len,
460  state->symmetric->ck, hash->hash_len, temp, hash->hash_len);
461  noise_symmetricstate_mix_hash(state->symmetric, temp, hash->hash_len);
462  noise_clean(temp, sizeof(temp));
463 
464  /* We have the pre shared key now */
465  state->requirements &= ~NOISE_REQ_PSK;
466  return NOISE_ERROR_NONE;
467 }

644 int noise_handshakestate_start(NoiseHandshakeState *state)
645 {
646  /* Validate the parameter */
647  if (!state)
648  return NOISE_ERROR_INVALID_PARAM;
649  if (state->action != NOISE_ACTION_NONE)
650  return NOISE_ERROR_INVALID_STATE;
651  if (state->symmetric->id.pattern_id == NOISE_PATTERN_XX_FALLBACK &&
652  (state->requirements & NOISE_REQ_FALLBACK_PREMSG) == 0)
653  return NOISE_ERROR_NOT_APPLICABLE;
654 
655  /* Check that we have satisfied all of the pattern requirements */
656  if ((state->requirements & NOISE_REQ_LOCAL_REQUIRED) != 0 &&
657  !noise_dhstate_has_keypair(state->dh_local_static))
658  return NOISE_ERROR_LOCAL_KEY_REQUIRED;
659  if ((state->requirements & NOISE_REQ_REMOTE_REQUIRED) != 0 &&
660  !noise_dhstate_has_public_key(state->dh_remote_static))
661  return NOISE_ERROR_REMOTE_KEY_REQUIRED;
662  if ((state->requirements & NOISE_REQ_PSK) != 0)
663  return NOISE_ERROR_PSK_REQUIRED;
664 
665  /* If the prologue has not been provided yet, hash an empty one */
666  if (state->requirements & NOISE_REQ_PROLOGUE)
667  noise_handshakestate_set_prologue(state, "", 0);
668 
669  /* Mix the pre-supplied public keys into the handshake hash */
670  if (state->role == NOISE_ROLE_INITIATOR) {
671  if (state->requirements & NOISE_REQ_LOCAL_PREMSG)
672  noise_handshakestate_mix_public_key(state, state->dh_local_static);
673  if (state->requirements & NOISE_REQ_REMOTE_PREMSG)
674  noise_handshakestate_mix_public_key(state, state->dh_remote_static);
675  if (state->requirements & NOISE_REQ_FALLBACK_PREMSG)
676  noise_handshakestate_mix_public_key(state, state->dh_remote_ephemeral);
677  } else {
678  if (state->requirements & NOISE_REQ_REMOTE_PREMSG)
679  noise_handshakestate_mix_public_key(state, state->dh_remote_static);
680  if (state->requirements & NOISE_REQ_LOCAL_PREMSG)
681  noise_handshakestate_mix_public_key(state, state->dh_local_static);
682  if (state->requirements & NOISE_REQ_FALLBACK_PREMSG)
683  noise_handshakestate_mix_public_key(state, state->dh_local_ephemeral);
684  }
685 
686  /* The handshake has now officially started */
687  if (state->role == NOISE_ROLE_INITIATOR)
688  state->action = NOISE_ACTION_WRITE_MESSAGE;
689  else
690  state->action = NOISE_ACTION_READ_MESSAGE;
691  return NOISE_ERROR_NONE;
692 }

726 int noise_handshakestate_fallback(NoiseHandshakeState *state)
727 {
728  char name[NOISE_MAX_PROTOCOL_NAME];
729  size_t hash_len;
730  size_t name_len;
731  NoiseProtocolId id;
732  int err;
733 
734  /* Validate the parameter */
735  if (!state)
736  return NOISE_ERROR_INVALID_PARAM;
737  if (state->symmetric->id.pattern_id != NOISE_PATTERN_IK)
738  return NOISE_ERROR_NOT_APPLICABLE;
739 
740  /* The initiator should be waiting for a return message from the
741  responder, and the responder should have failed on the first
742  handshake message from the initiator. We also allow the
743  responder to fallback after processing the first message
744  successfully; it decides to always fall back anyway. */
745  if (state->role == NOISE_ROLE_INITIATOR) {
746  if (state->action != NOISE_ACTION_READ_MESSAGE)
747  return NOISE_ERROR_INVALID_STATE;
748  if (!noise_dhstate_has_public_key(state->dh_local_ephemeral))
749  return NOISE_ERROR_INVALID_STATE;
750  } else {
751  if (state->action != NOISE_ACTION_FAILED &&
752  state->action != NOISE_ACTION_WRITE_MESSAGE)
753  return NOISE_ERROR_INVALID_STATE;
754  if (!noise_dhstate_has_public_key(state->dh_remote_ephemeral))
755  return NOISE_ERROR_INVALID_STATE;
756  }
757 
758  /* Format a new protocol name for the "XXfallback" variant */
759  id = state->symmetric->id;
760  id.pattern_id = NOISE_PATTERN_XX_FALLBACK;
761  err = noise_protocol_id_to_name(name, sizeof(name), &id);
762  if (err != NOISE_ERROR_NONE)
763  return err;
764 
765  /* Convert the HandshakeState to the "XXfallback" pattern */
766  state->symmetric->id.pattern_id = NOISE_PATTERN_XX_FALLBACK;
767  noise_dhstate_clear_key(state->dh_remote_static);
768  if (state->role == NOISE_ROLE_INITIATOR) {
769  noise_dhstate_clear_key(state->dh_remote_ephemeral);
770  state->action = NOISE_ACTION_READ_MESSAGE;
771  state->role = NOISE_ROLE_RESPONDER;
772  } else {
773  noise_dhstate_clear_key(state->dh_local_ephemeral);
774  state->action = NOISE_ACTION_WRITE_MESSAGE;
775  state->role = NOISE_ROLE_INITIATOR;
776  }
777  state->requirements |= NOISE_REQ_FALLBACK_PREMSG;
778 
779  /* Re-initialize the chaining key "ck" and the handshake hash "h" from
780  the new protocol name. If the name is too long, hash it down first */
781  name_len = strlen(name);
782  hash_len = noise_hashstate_get_hash_length(state->symmetric->hash);
783  if (name_len <= hash_len) {
784  memcpy(state->symmetric->h, name, name_len);
785  memset(state->symmetric->h + name_len, 0, hash_len - name_len);
786  } else {
787  noise_hashstate_hash_one
788  (state->symmetric->hash, (const uint8_t *)name, name_len,
789  state->symmetric->h, hash_len);
790  }
791  memcpy(state->symmetric->ck, state->symmetric->h, hash_len);
792 
793  /* Reset the encryption key within the symmetric state to empty */
794  state->symmetric->cipher->has_key = 0;
795  state->symmetric->cipher->n = 0;
796  state->symmetric->cipher->nonce_overflow = 0;
797 
798  /* Ready to go */
799  return NOISE_ERROR_NONE;
800 }

870 static int noise_handshakestate_write
871  (NoiseHandshakeState *state, NoiseBuffer *message, const NoiseBuffer *payload)
872 {
873  NoiseBuffer rest;
874  size_t len;
875  size_t mac_len;
876  uint8_t token;
877  int err;
878 
879  /* Process tokens until the direction changes or the pattern ends */
880  for (;;) {
881  token = *(state->tokens);
882  if (token == NOISE_TOKEN_END) {
883  /* The pattern has finished, so the next action is "split" */
884  state->action = NOISE_ACTION_SPLIT;
885  break;
886  } else if (token == NOISE_TOKEN_FLIP_DIR) {
887  /* Changing directions, so this message is complete and
888  the next action is "read message". */
889  ++(state->tokens);
890  state->action = NOISE_ACTION_READ_MESSAGE;
891  break;
892  }
893 
894  /* Set "rest" to the rest of the "message" buffer after the
895  current size. This is the space we have left to write
896  handshake values while processing this token. */
897  rest.data = message->data + message->size;
898  rest.size = 0;
899  rest.max_size = message->max_size - message->size;
900 
901  /* Process the token */
902  err = NOISE_ERROR_NONE;
903  switch (token) {
904  case NOISE_TOKEN_E:
905  /* Generate a local ephemeral keypair and add the public
906  key to the message. If we are running fixed vector tests,
907  then the ephemeral key may have already been provided. */
908  if (!state->dh_local_ephemeral)
909  return NOISE_ERROR_INVALID_STATE;
910  len = state->dh_local_ephemeral->public_key_len;
911  if (!state->dh_fixed_ephemeral) {
912  err = noise_dhstate_generate_keypair(state->dh_local_ephemeral);
913  } else {
914  /* Use the fixed ephemeral key provided by the test harness */
915  err = noise_dhstate_set_keypair
916  (state->dh_local_ephemeral,
917  state->dh_fixed_ephemeral->private_key,
918  state->dh_fixed_ephemeral->private_key_len,
919  state->dh_fixed_ephemeral->public_key,
920  state->dh_fixed_ephemeral->public_key_len);
921  }
922  if (err != NOISE_ERROR_NONE)
923  break;
924  if (rest.max_size < len)
925  return NOISE_ERROR_INVALID_LENGTH;
926  memcpy(rest.data, state->dh_local_ephemeral->public_key, len);
927  noise_symmetricstate_mix_hash(state->symmetric, rest.data, len);
928  rest.size += len;
929 
930  /* If the protocol is using pre-shared keys, then also mix
931  the local ephemeral key into the chaining key */
932  if (state->symmetric->id.prefix_id == NOISE_PREFIX_PSK) {
933  err = noise_symmetricstate_mix_key
934  (state->symmetric,
935  state->dh_local_ephemeral->public_key, len);
936  }
937  break;
938  case NOISE_TOKEN_S:
939  /* Encrypt the local static public key and add it to the message */
940  if (!state->dh_local_static)
941  return NOISE_ERROR_INVALID_STATE;
942  len = state->dh_local_static->public_key_len;
943  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
944  if (rest.max_size < (len + mac_len))
945  return NOISE_ERROR_INVALID_LENGTH;
946  memcpy(rest.data, state->dh_local_static->public_key, len);
947  rest.size += len;
948  err = noise_symmetricstate_encrypt_and_hash(state->symmetric, &rest);
949  if (err != NOISE_ERROR_NONE)
950  break;
951  break;
952  case NOISE_TOKEN_DHEE:
953  /* DH operation with local and remote ephemeral keys */
954  err = noise_handshake_mix_dh
955  (state, state->dh_local_ephemeral, state->dh_remote_ephemeral);
956  break;
957  case NOISE_TOKEN_DHES:
958  /* DH operation with local ephemeral and remote static keys */
959  err = noise_handshake_mix_dh
960  (state, state->dh_local_ephemeral, state->dh_remote_static);
961  break;
962  case NOISE_TOKEN_DHSE:
963  /* DH operation with local static and remote ephemeral keys */
964  err = noise_handshake_mix_dh
965  (state, state->dh_local_static, state->dh_remote_ephemeral);
966  break;
967  case NOISE_TOKEN_DHSS:
968  /* DH operation with local and remote static keys */
969  err = noise_handshake_mix_dh
970  (state, state->dh_local_static, state->dh_remote_static);
971  break;
972  default:
973  /* Unknown token code in the pattern. This shouldn't happen.
974  If it does, then abort immediately. */
975  err = NOISE_ERROR_INVALID_STATE;
976  break;
977  }
978  if (err != NOISE_ERROR_NONE)
979  return err;
980  message->size += rest.size;
981  ++(state->tokens);

993  rest.max_size = message->max_size - message->size;
994  memcpy(rest.data, payload->data, payload->size);
995  } else {

997  rest.size = 0;
998  rest.max_size = message->max_size - message->size;
999  }

1001  if (err != NOISE_ERROR_NONE)
1002  return err;
1003 
1004  /* Return the final size to the caller */
1005  message->size += rest.size;
1006  return NOISE_ERROR_NONE;

1085 static int noise_handshakestate_read
1086  (NoiseHandshakeState *state, NoiseBuffer *message, NoiseBuffer *payload)
1087 {
1088  NoiseBuffer msg;
1089  NoiseBuffer msg2;
1090  size_t len;
1091  size_t mac_len;
1092  uint8_t token;
1093  int err;
1094 
1095  /* Make a copy of the message buffer. As we process tokens, the copy
1096  will become shorter and shorter until only the payload is left. */
1097  msg = *message;
1098 
1099  /* Process tokens until the direction changes or the pattern ends */
1100  for (;;) {
1101  token = *(state->tokens);
1102  if (token == NOISE_TOKEN_END) {
1103  /* The pattern has finished, so the next action is "split" */
1104  state->action = NOISE_ACTION_SPLIT;
1105  break;
1106  } else if (token == NOISE_TOKEN_FLIP_DIR) {
1107  /* Changing directions, so this message is complete and
1108  the next action is "read message". */
1109  ++(state->tokens);
1110  state->action = NOISE_ACTION_WRITE_MESSAGE;
1111  break;
1112  }
1113  err = NOISE_ERROR_NONE;
1114  switch (token) {
1115  case NOISE_TOKEN_E:
1116  /* Save the remote ephemeral key and hash it */
1117  if (!state->dh_remote_ephemeral)
1118  return NOISE_ERROR_INVALID_STATE;
1119  len = state->dh_remote_ephemeral->public_key_len;
1120  if (msg.size < len)
1121  return NOISE_ERROR_INVALID_LENGTH;
1122  err = noise_symmetricstate_mix_hash
1123  (state->symmetric, msg.data, len);
1124  if (err != NOISE_ERROR_NONE)
1125  break;
1126  err = noise_dhstate_set_public_key
1127  (state->dh_remote_ephemeral, msg.data, len);
1128  if (err != NOISE_ERROR_NONE)
1129  break;
1130  if (noise_dhstate_is_null_public_key(state->dh_remote_ephemeral)) {
1131  /* The remote ephemeral key is null, which means that it is
1132  not contributing anything to the security of the session
1133  and is in fact downgrading the security to "none at all"
1134  in some of the message patterns. Reject all such keys. */
1135  return NOISE_ERROR_INVALID_PUBLIC_KEY;
1136  }
1137  msg.data += len;
1138  msg.size -= len;
1139  msg.max_size -= len;
1140 
1141  /* If the protocol is using pre-shared keys, then also mix
1142  the remote ephemeral key into the chaining key */
1143  if (state->symmetric->id.prefix_id == NOISE_PREFIX_PSK) {
1144  err = noise_symmetricstate_mix_key
1145  (state->symmetric,
1146  state->dh_remote_ephemeral->public_key, len);
1147  }
1148  break;
1149  case NOISE_TOKEN_S:
1150  /* Decrypt and read the remote static key */
1151  if (!state->dh_remote_static)
1152  return NOISE_ERROR_INVALID_STATE;
1153  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
1154  len = state->dh_remote_static->public_key_len + mac_len;
1155  if (msg.size < len)
1156  return NOISE_ERROR_INVALID_LENGTH;
1157  msg2.data = msg.data;
1158  msg2.size = len;
1159  msg2.max_size = len;
1160  err = noise_symmetricstate_decrypt_and_hash
1161  (state->symmetric, &msg2);
1162  if (err != NOISE_ERROR_NONE)
1163  break;
1164  err = noise_dhstate_set_public_key
1165  (state->dh_remote_static, msg2.data, msg2.size);
1166  if (err != NOISE_ERROR_NONE)
1167  break;
1168  msg.data += len;
1169  msg.size -= len;
1170  msg.max_size -= len;
1171  break;
1172  case NOISE_TOKEN_DHEE:
1173  /* DH operation with local and remote ephemeral keys */
1174  err = noise_handshake_mix_dh
1175  (state, state->dh_local_ephemeral, state->dh_remote_ephemeral);
1176  break;
1177  case NOISE_TOKEN_DHES:
1178  /* DH operation with remote ephemeral and local static keys */
1179  err = noise_handshake_mix_dh
1180  (state, state->dh_local_static, state->dh_remote_ephemeral);
1181  break;
1182  case NOISE_TOKEN_DHSE:
1183  /* DH operation with remote static and local ephemeral keys */
1184  err = noise_handshake_mix_dh
1185  (state, state->dh_local_ephemeral, state->dh_remote_static);
1186  break;
1187  case NOISE_TOKEN_DHSS:
1188  /* DH operation with local and remote static keys */
1189  err = noise_handshake_mix_dh
1190  (state, state->dh_local_static, state->dh_remote_static);
1191  break;
1192  default:
1193  /* Unknown token code in the pattern. This shouldn't happen.
1194  If it does, then abort immediately. */
1195  err = NOISE_ERROR_INVALID_STATE;
1196  break;
1197  }
1198  if (err != NOISE_ERROR_NONE)
1199  return err;
1200  ++(state->tokens);
1201  }
1202 
1203  /* Decrypt the remaining bytes and return them in the payload buffer */
1204  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
1205  err = noise_symmetricstate_decrypt_and_hash(state->symmetric, &msg);
1206  if (err != NOISE_ERROR_NONE)
1207  return err;
1208  if (payload) {
1209  if (msg.size > payload->max_size)
1210  return NOISE_ERROR_INVALID_LENGTH;
1211  memcpy(payload->data, msg.data, msg.size);
1212  payload->size = msg.size;
1213  }
1214  return NOISE_ERROR_NONE;
1215 }
