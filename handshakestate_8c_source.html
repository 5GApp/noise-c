60 static int noise_handshakestate_requirements
61  (uint8_t flags, int prefix_id, int role, int is_fallback)
62 {
63  int requirements = 0;
64  if (flags & NOISE_PAT_FLAG_LOCAL_STATIC) {
65  requirements |= NOISE_REQ_LOCAL_REQUIRED;
66  }
67  if (flags & NOISE_PAT_FLAG_LOCAL_REQUIRED) {
68  requirements |= NOISE_REQ_LOCAL_REQUIRED;
69  requirements |= NOISE_REQ_LOCAL_PREMSG;
70  }
71  if (flags & NOISE_PAT_FLAG_REMOTE_REQUIRED) {
72  requirements |= NOISE_REQ_REMOTE_REQUIRED;
73  requirements |= NOISE_REQ_REMOTE_PREMSG;
74  }
75  if (flags & (NOISE_PAT_FLAG_REMOTE_EPHEM_REQ |
76  NOISE_PAT_FLAG_LOCAL_EPHEM_REQ)) {
77  if (is_fallback)
78  requirements |= NOISE_REQ_FALLBACK_PREMSG;
79  }
80  if (prefix_id == NOISE_PREFIX_PSK) {
81  requirements |= NOISE_REQ_PSK;
82  }
83  return requirements;
84 }

102 static int noise_handshakestate_new
103  (NoiseHandshakeState **state, NoiseSymmetricState *symmetric, int role)
104 {
105  const uint8_t *pattern;
106  int dh_id;
107  uint8_t flags;
108  int extra_reqs = 0;
109  int err;
110  int local_dh_role;
111  int remote_dh_role;
112 
113  /* Locate the information for the current handshake pattern */
114  pattern = noise_pattern_lookup(symmetric->id.pattern_id);
115  if (!pattern) {
116  noise_symmetricstate_free(symmetric);
117  return NOISE_ERROR_UNKNOWN_ID;
118  }
119  flags = pattern[0];
120  if ((flags & NOISE_PAT_FLAG_REMOTE_REQUIRED) != 0)
121  extra_reqs |= NOISE_REQ_FALLBACK_POSSIBLE;
122  if (role == NOISE_ROLE_RESPONDER) {
123  /* Reverse the pattern flags so that the responder is "local" */
124  flags = noise_pattern_reverse_flags(flags);
125  }
126 
127  /* Determine the role to use for DHState objects */
128  if ((flags & NOISE_PAT_FLAG_REMOTE_EPHEM_REQ) != 0) {
129  /* Fallback pattern - reverse the DHState role */
130  if (role == NOISE_ROLE_INITIATOR) {
131  local_dh_role = NOISE_ROLE_RESPONDER;
132  remote_dh_role = NOISE_ROLE_INITIATOR;
133  } else {
134  local_dh_role = NOISE_ROLE_INITIATOR;
135  remote_dh_role = NOISE_ROLE_RESPONDER;
136  }
137  } else {
138  /* Regular pattern */
139  if (role == NOISE_ROLE_INITIATOR) {
140  local_dh_role = NOISE_ROLE_INITIATOR;
141  remote_dh_role = NOISE_ROLE_RESPONDER;
142  } else {
143  local_dh_role = NOISE_ROLE_RESPONDER;
144  remote_dh_role = NOISE_ROLE_INITIATOR;
145  }
146  }
147 
148  /* Create the HandshakeState object */
149  *state = noise_new(NoiseHandshakeState);
150  if (!(*state)) {
151  noise_symmetricstate_free(symmetric);
152  return NOISE_ERROR_NO_MEMORY;
153  }
154 
155  /* Initialize the HandshakeState */
156  (*state)->requirements = extra_reqs | noise_handshakestate_requirements
157  (flags, symmetric->id.prefix_id, role, 0);
158  (*state)->action = NOISE_ACTION_NONE;
159  (*state)->tokens = pattern + 1;
160  (*state)->role = role;
161  (*state)->symmetric = symmetric;
162 
163  /* Create DHState objects for all of the keys we will need later */
164  err = NOISE_ERROR_NONE;
165  dh_id = symmetric->id.dh_id;
166  if ((flags & NOISE_PAT_FLAG_LOCAL_STATIC) != 0)
167  err = noise_dhstate_new_by_id(&((*state)->dh_local_static), dh_id);
168  if ((flags & NOISE_PAT_FLAG_LOCAL_EPHEMERAL) != 0 && err == NOISE_ERROR_NONE)
169  err = noise_dhstate_new_by_id(&((*state)->dh_local_ephemeral), dh_id);
170  if ((flags & NOISE_PAT_FLAG_REMOTE_STATIC) != 0 && err == NOISE_ERROR_NONE)
171  err = noise_dhstate_new_by_id(&((*state)->dh_remote_static), dh_id);
172  if ((flags & NOISE_PAT_FLAG_REMOTE_EPHEMERAL) != 0 && err == NOISE_ERROR_NONE)
173  err = noise_dhstate_new_by_id(&((*state)->dh_remote_ephemeral), dh_id);
174 
175  /* Set the roles for the DHState objects */
176  noise_dhstate_set_role((*state)->dh_local_ephemeral, local_dh_role);
177  noise_dhstate_set_role((*state)->dh_local_static, local_dh_role);
178  noise_dhstate_set_role((*state)->dh_remote_ephemeral, remote_dh_role);
179  noise_dhstate_set_role((*state)->dh_remote_static, remote_dh_role);
180 
181  /* If the DH algorithm is ephemeral-only, then we need to apply some
182  * extra checks on the pattern: essentially, only "NN" is possible */
183  if (err == NOISE_ERROR_NONE) {
184  if ((*state)->dh_local_static && (*state)->dh_local_static->ephemeral_only)
185  err = NOISE_ERROR_NOT_APPLICABLE;
186  if ((*state)->dh_remote_static && (*state)->dh_remote_static->ephemeral_only)
187  err = NOISE_ERROR_NOT_APPLICABLE;
188  if ((*state)->dh_local_ephemeral && (*state)->dh_local_ephemeral->ephemeral_only) {
189  if (!((*state)->dh_remote_ephemeral))
190  err = NOISE_ERROR_NOT_APPLICABLE;
191  } else if ((*state)->dh_remote_ephemeral && (*state)->dh_remote_ephemeral->ephemeral_only) {
192  err = NOISE_ERROR_NOT_APPLICABLE;
193  }
194  }
195 
196  /* Bail out if we had an error trying to create the DHState objects */
197  if (err != NOISE_ERROR_NONE) {
198  noise_handshakestate_free(*state);
199  return err;
200  }
201 
202  /* Ready to go */
203  return NOISE_ERROR_NONE;
204 }

310 {

725 int noise_handshakestate_start(NoiseHandshakeState *state)
726 {
727  /* Validate the parameter */
728  if (!state)
729  return NOISE_ERROR_INVALID_PARAM;
730  if (state->action != NOISE_ACTION_NONE)
731  return NOISE_ERROR_INVALID_STATE;
732  if (state->symmetric->id.pattern_id == NOISE_PATTERN_XX_FALLBACK &&
733  (state->requirements & NOISE_REQ_FALLBACK_PREMSG) == 0)
734  return NOISE_ERROR_NOT_APPLICABLE;
735 
736  /* Check that we have satisfied all of the pattern requirements */
737  if ((state->requirements & NOISE_REQ_LOCAL_REQUIRED) != 0 &&
738  !noise_dhstate_has_keypair(state->dh_local_static))
739  return NOISE_ERROR_LOCAL_KEY_REQUIRED;
740  if ((state->requirements & NOISE_REQ_REMOTE_REQUIRED) != 0 &&
741  !noise_dhstate_has_public_key(state->dh_remote_static))
742  return NOISE_ERROR_REMOTE_KEY_REQUIRED;
743  if ((state->requirements & NOISE_REQ_PSK) != 0 &&
744  state->pre_shared_key_len == 0)
745  return NOISE_ERROR_PSK_REQUIRED;
746 
747  /* Hash the prologue value */
748  if (state->prologue_len) {
749  noise_symmetricstate_mix_hash
750  (state->symmetric, state->prologue, state->prologue_len);
751  } else {
752  /* No prologue, so hash an empty one */
753  noise_symmetricstate_mix_hash
754  (state->symmetric, state->pre_shared_key, 0);
755  }
756 
757  /* Mix the pre shared key into the chaining key and handshake hash */
758  if (state->pre_shared_key_len) {
759  uint8_t temp[NOISE_MAX_HASHLEN];
760  NoiseHashState *hash = state->symmetric->hash;
761  noise_hashstate_hkdf
762  (hash, state->symmetric->ck, hash->hash_len,
763  state->pre_shared_key, state->pre_shared_key_len,
764  state->symmetric->ck, hash->hash_len, temp, hash->hash_len);
765  noise_symmetricstate_mix_hash(state->symmetric, temp, hash->hash_len);
766  noise_clean(temp, sizeof(temp));
767  }
768 
769  /* Mix the pre-supplied public keys into the handshake hash */
770  if (state->role == NOISE_ROLE_INITIATOR) {
771  if (state->requirements & NOISE_REQ_LOCAL_PREMSG)
772  noise_handshakestate_mix_public_key(state, state->dh_local_static);
773  if (state->requirements & NOISE_REQ_FALLBACK_PREMSG) {
774  noise_handshakestate_mix_public_key(state, state->dh_remote_ephemeral);
775  if ((state->requirements & NOISE_REQ_PSK) != 0) {
776  noise_handshakestate_mix_chaining_key
777  (state, state->dh_remote_ephemeral);
778  }
779  }
780  if (state->requirements & NOISE_REQ_REMOTE_PREMSG)
781  noise_handshakestate_mix_public_key(state, state->dh_remote_static);
782  } else {
783  if (state->requirements & NOISE_REQ_REMOTE_PREMSG)
784  noise_handshakestate_mix_public_key(state, state->dh_remote_static);
785  if (state->requirements & NOISE_REQ_FALLBACK_PREMSG) {
786  noise_handshakestate_mix_public_key(state, state->dh_local_ephemeral);
787  if ((state->requirements & NOISE_REQ_PSK) != 0) {
788  noise_handshakestate_mix_chaining_key
789  (state, state->dh_local_ephemeral);
790  }
791  }
792  if (state->requirements & NOISE_REQ_LOCAL_PREMSG)
793  noise_handshakestate_mix_public_key(state, state->dh_local_static);
794  }
795 
796  /* The handshake has now officially started */
797  if (state->role == NOISE_ROLE_INITIATOR)
798  state->action = NOISE_ACTION_WRITE_MESSAGE;
799  else
800  state->action = NOISE_ACTION_READ_MESSAGE;
801  return NOISE_ERROR_NONE;
802 }

890 int noise_handshakestate_fallback_to(NoiseHandshakeState *state, int pattern_id)
891 {
892  char name[NOISE_MAX_PROTOCOL_NAME];
893  size_t hash_len;
894  size_t name_len;
895  NoiseProtocolId id;
896  const uint8_t *pattern;
897  uint8_t flags;
898  int err;
899 
900  /* Validate the parameter */
901  if (!state)
902  return NOISE_ERROR_INVALID_PARAM;
903 
904  /* The original pattern must end in "K" for fallback to be possible */
905  if (state->symmetric->id.pattern_id < NOISE_PATTERN_NN ||
906  (state->requirements & NOISE_REQ_FALLBACK_POSSIBLE) == 0)
907  return NOISE_ERROR_NOT_APPLICABLE;
908 
909  /* Check that "pattern_id" supports fallback */
910  pattern = noise_pattern_lookup(pattern_id);
911  if (!pattern || (*pattern & NOISE_PAT_FLAG_REMOTE_EPHEM_REQ) == 0)
912  return NOISE_ERROR_NOT_APPLICABLE;
913 
914  /* The initiator should be waiting for a return message from the
915  responder, and the responder should have failed on the first
916  handshake message from the initiator. We also allow the
917  responder to fallback after processing the first message
918  successfully; it decides to always fall back anyway. */
919  if (state->role == NOISE_ROLE_INITIATOR) {
920  if (state->action != NOISE_ACTION_FAILED &&
921  state->action != NOISE_ACTION_READ_MESSAGE)
922  return NOISE_ERROR_INVALID_STATE;
923  if (!noise_dhstate_has_public_key(state->dh_local_ephemeral))
924  return NOISE_ERROR_INVALID_STATE;
925  } else {
926  if (state->action != NOISE_ACTION_FAILED &&
927  state->action != NOISE_ACTION_WRITE_MESSAGE)
928  return NOISE_ERROR_INVALID_STATE;
929  if (!noise_dhstate_has_public_key(state->dh_remote_ephemeral))
930  return NOISE_ERROR_INVALID_STATE;
931  }
932 
933  /* Format a new protocol name for the fallback variant */
934  id = state->symmetric->id;
935  id.pattern_id = pattern_id;
936  err = noise_protocol_id_to_name(name, sizeof(name), &id);
937  if (err != NOISE_ERROR_NONE)
938  return err;
939 
940  /* Convert the HandshakeState to the fallback pattern */
941  state->symmetric->id.pattern_id = pattern_id;
942  if (state->role == NOISE_ROLE_INITIATOR) {
943  noise_dhstate_clear_key(state->dh_remote_ephemeral);
944  noise_dhstate_clear_key(state->dh_remote_static);
945  state->role = NOISE_ROLE_RESPONDER;
946  } else {
947  noise_dhstate_clear_key(state->dh_local_ephemeral);
948  if (!(pattern[0] & NOISE_PAT_FLAG_REMOTE_REQUIRED))
949  noise_dhstate_clear_key(state->dh_remote_static);
950  state->role = NOISE_ROLE_INITIATOR;
951  }
952 
953  /* Start a new token pattern for the fallback */
954  state->tokens = pattern + 1;
955  state->action = NOISE_ACTION_NONE;
956 
957  /* Set up the key requirements for the fallback */
958  flags = pattern[0];
959  if (state->role == NOISE_ROLE_RESPONDER) {
960  flags = noise_pattern_reverse_flags(flags);
961  }
962  state->requirements = noise_handshakestate_requirements
963  (flags, id.prefix_id, state->role, 1);
964 
965  /* Re-initialize the chaining key "ck" and the handshake hash "h" from
966  the new protocol name. If the name is too long, hash it down first */
967  name_len = strlen(name);
968  hash_len = noise_hashstate_get_hash_length(state->symmetric->hash);
969  if (name_len <= hash_len) {
970  memcpy(state->symmetric->h, name, name_len);
971  memset(state->symmetric->h + name_len, 0, hash_len - name_len);
972  } else {
973  noise_hashstate_hash_one
974  (state->symmetric->hash, (const uint8_t *)name, name_len,
975  state->symmetric->h, hash_len);
976  }
977  memcpy(state->symmetric->ck, state->symmetric->h, hash_len);
978 
979  /* Reset the encryption key within the symmetric state to empty */

1057 static int noise_handshakestate_write
1058  (NoiseHandshakeState *state, NoiseBuffer *message, const NoiseBuffer *payload)
1059 {
1060  NoiseBuffer rest;
1061  size_t len;
1062  size_t mac_len;
1063  uint8_t token;
1064  int err;
1065 
1066  /* Process tokens until the direction changes or the pattern ends */
1067  for (;;) {
1068  token = *(state->tokens);
1069  if (token == NOISE_TOKEN_END) {
1070  /* The pattern has finished, so the next action is "split" */
1071  state->action = NOISE_ACTION_SPLIT;
1072  break;
1073  } else if (token == NOISE_TOKEN_FLIP_DIR) {
1074  /* Changing directions, so this message is complete and
1075  the next action is "read message". */
1076  ++(state->tokens);
1077  state->action = NOISE_ACTION_READ_MESSAGE;
1078  break;
1079  }
1080 
1081  /* Set "rest" to the rest of the "message" buffer after the
1082  current size. This is the space we have left to write
1083  handshake values while processing this token. */
1084  rest.data = message->data + message->size;
1085  rest.size = 0;
1086  rest.max_size = message->max_size - message->size;
1087 
1088  /* Process the token */
1089  err = NOISE_ERROR_NONE;
1090  switch (token) {
1091  case NOISE_TOKEN_E:
1092  /* Generate a local ephemeral keypair and add the public
1093  key to the message. If we are running fixed vector tests,
1094  then the ephemeral key may have already been provided. */
1095  if (!state->dh_local_ephemeral)
1096  return NOISE_ERROR_INVALID_STATE;
1097  if (!state->dh_fixed_ephemeral) {
1098  err = noise_dhstate_generate_dependent_keypair
1099  (state->dh_local_ephemeral, state->dh_remote_ephemeral);
1100  } else {
1101  /* Use the fixed ephemeral key provided by the test harness */
1102  err = noise_dhstate_set_keypair
1103  (state->dh_local_ephemeral,
1104  state->dh_fixed_ephemeral->private_key,
1105  state->dh_fixed_ephemeral->private_key_len,
1106  state->dh_fixed_ephemeral->public_key,
1107  state->dh_fixed_ephemeral->public_key_len);
1108  }
1109  if (err != NOISE_ERROR_NONE)
1110  break;
1111  len = state->dh_local_ephemeral->public_key_len;
1112  if (rest.max_size < len)
1113  return NOISE_ERROR_INVALID_LENGTH;
1114  memcpy(rest.data, state->dh_local_ephemeral->public_key, len);
1115  noise_symmetricstate_mix_hash(state->symmetric, rest.data, len);
1116  rest.size += len;
1117 
1118  /* If the protocol is using pre-shared keys, then also mix
1119  the local ephemeral key into the chaining key */
1120  if (state->symmetric->id.prefix_id == NOISE_PREFIX_PSK) {
1121  err = noise_symmetricstate_mix_key
1122  (state->symmetric,
1123  state->dh_local_ephemeral->public_key, len);
1124  }
1125  break;
1126  case NOISE_TOKEN_S:
1127  /* Encrypt the local static public key and add it to the message */
1128  if (!state->dh_local_static)
1129  return NOISE_ERROR_INVALID_STATE;
1130  len = state->dh_local_static->public_key_len;
1131  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
1132  if (rest.max_size < (len + mac_len))
1133  return NOISE_ERROR_INVALID_LENGTH;
1134  memcpy(rest.data, state->dh_local_static->public_key, len);
1135  rest.size += len;
1136  err = noise_symmetricstate_encrypt_and_hash(state->symmetric, &rest);
1137  if (err != NOISE_ERROR_NONE)
1138  break;
1139  break;
1140  case NOISE_TOKEN_DHEE:
1141  /* DH operation with local and remote ephemeral keys */
1142  err = noise_handshake_mix_dh
1143  (state, state->dh_local_ephemeral, state->dh_remote_ephemeral);
1144  break;
1145  case NOISE_TOKEN_DHES:
1146  /* DH operation with local ephemeral and remote static keys */
1147  err = noise_handshake_mix_dh
1148  (state, state->dh_local_ephemeral, state->dh_remote_static);
1149  break;
1150  case NOISE_TOKEN_DHSE:
1151  /* DH operation with local static and remote ephemeral keys */
1152  err = noise_handshake_mix_dh
1153  (state, state->dh_local_static, state->dh_remote_ephemeral);
1154  break;
1155  case NOISE_TOKEN_DHSS:
1156  /* DH operation with local and remote static keys */
1157  err = noise_handshake_mix_dh
1158  (state, state->dh_local_static, state->dh_remote_static);
1159  break;
1160  default:
1161  /* Unknown token code in the pattern. This shouldn't happen.
1162  If it does, then abort immediately. */
1163  err = NOISE_ERROR_INVALID_STATE;
1164  break;
1165  }
1166  if (err != NOISE_ERROR_NONE)
1167  return err;
1168  message->size += rest.size;
1169  ++(state->tokens);
1170  }
1171 
1172  /* Add the payload to the message buffer and encrypt it */
1173  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
1174  if ((message->max_size - message->size) < mac_len)
1175  return NOISE_ERROR_INVALID_LENGTH;
1176  if (payload) {
1177  if ((message->max_size - message->size - mac_len) < payload->size)
1178  return NOISE_ERROR_INVALID_LENGTH;
1179  rest.data = message->data + message->size;
1180  rest.size = payload->size;
1181  rest.max_size = message->max_size - message->size;
1182  memcpy(rest.data, payload->data, payload->size);
1183  } else {
1184  rest.data = message->data + message->size;
1185  rest.size = 0;
1186  rest.max_size = message->max_size - message->size;
1187  }
1188  err = noise_symmetricstate_encrypt_and_hash(state->symmetric, &rest);
1189  if (err != NOISE_ERROR_NONE)
1190  return err;
1191 
1192  /* Return the final size to the caller */
1193  message->size += rest.size;
1194  return NOISE_ERROR_NONE;
1195 }

1273 static int noise_handshakestate_read
1274  (NoiseHandshakeState *state, NoiseBuffer *message, NoiseBuffer *payload)
1275 {
1276  NoiseBuffer msg;
1277  NoiseBuffer msg2;
1278  size_t len;
1279  size_t mac_len;
1280  uint8_t token;
1281  int err;
1282 
1283  /* Make a copy of the message buffer. As we process tokens, the copy
1284  will become shorter and shorter until only the payload is left. */
1285  msg = *message;
1286 
1287  /* Process tokens until the direction changes or the pattern ends */
1288  for (;;) {
1289  token = *(state->tokens);
1290  if (token == NOISE_TOKEN_END) {
1291  /* The pattern has finished, so the next action is "split" */
1292  state->action = NOISE_ACTION_SPLIT;
1293  break;
1294  } else if (token == NOISE_TOKEN_FLIP_DIR) {
1295  /* Changing directions, so this message is complete and
1296  the next action is "read message". */
1297  ++(state->tokens);
1298  state->action = NOISE_ACTION_WRITE_MESSAGE;
1299  break;
1300  }
1301  err = NOISE_ERROR_NONE;
1302  switch (token) {
1303  case NOISE_TOKEN_E:
1304  /* Save the remote ephemeral key and hash it */
1305  if (!state->dh_remote_ephemeral)
1306  return NOISE_ERROR_INVALID_STATE;
1307  len = state->dh_remote_ephemeral->public_key_len;
1308  if (msg.size < len)
1309  return NOISE_ERROR_INVALID_LENGTH;
1310  err = noise_symmetricstate_mix_hash
1311  (state->symmetric, msg.data, len);
1312  if (err != NOISE_ERROR_NONE)
1313  break;
1314  err = noise_dhstate_set_public_key
1315  (state->dh_remote_ephemeral, msg.data, len);
1316  if (err != NOISE_ERROR_NONE)
1317  break;
1318  if (noise_dhstate_is_null_public_key(state->dh_remote_ephemeral)) {
1319  /* The remote ephemeral key is null, which means that it is
1320  not contributing anything to the security of the session
1321  and is in fact downgrading the security to "none at all"
1322  in some of the message patterns. Reject all such keys. */
1323  return NOISE_ERROR_INVALID_PUBLIC_KEY;
1324  }
1325  msg.data += len;
1326  msg.size -= len;
1327  msg.max_size -= len;
1328 
1329  /* If the protocol is using pre-shared keys, then also mix
1330  the remote ephemeral key into the chaining key */
1331  if (state->symmetric->id.prefix_id == NOISE_PREFIX_PSK) {
1332  err = noise_symmetricstate_mix_key
1333  (state->symmetric,
1334  state->dh_remote_ephemeral->public_key, len);
1335  }
1336  break;
1337  case NOISE_TOKEN_S:
1338  /* Decrypt and read the remote static key */
1339  if (!state->dh_remote_static)
1340  return NOISE_ERROR_INVALID_STATE;
1341  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
1342  len = state->dh_remote_static->public_key_len + mac_len;
1343  if (msg.size < len)
1344  return NOISE_ERROR_INVALID_LENGTH;
1345  msg2.data = msg.data;
1346  msg2.size = len;
1347  msg2.max_size = len;
1348  err = noise_symmetricstate_decrypt_and_hash
1349  (state->symmetric, &msg2);
1350  if (err != NOISE_ERROR_NONE)
1351  break;
1352  err = noise_dhstate_set_public_key
1353  (state->dh_remote_static, msg2.data, msg2.size);
1354  if (err != NOISE_ERROR_NONE)
1355  break;
1356  msg.data += len;
1357  msg.size -= len;
1358  msg.max_size -= len;
1359  break;
1360  case NOISE_TOKEN_DHEE:
1361  /* DH operation with local and remote ephemeral keys */
1362  err = noise_handshake_mix_dh
1363  (state, state->dh_local_ephemeral, state->dh_remote_ephemeral);
1364  break;
1365  case NOISE_TOKEN_DHES:
1366  /* DH operation with remote ephemeral and local static keys */
1367  err = noise_handshake_mix_dh
1368  (state, state->dh_local_static, state->dh_remote_ephemeral);
1369  break;
1370  case NOISE_TOKEN_DHSE:
1371  /* DH operation with remote static and local ephemeral keys */
1372  err = noise_handshake_mix_dh
1373  (state, state->dh_local_ephemeral, state->dh_remote_static);
1374  break;
1375  case NOISE_TOKEN_DHSS:
1376  /* DH operation with local and remote static keys */
1377  err = noise_handshake_mix_dh
1378  (state, state->dh_local_static, state->dh_remote_static);
1379  break;
1380  default:
1381  /* Unknown token code in the pattern. This shouldn't happen.
1382  If it does, then abort immediately. */
1383  err = NOISE_ERROR_INVALID_STATE;
1384  break;
1385  }
1386  if (err != NOISE_ERROR_NONE)
1387  return err;
1388  ++(state->tokens);
1389  }
1390 
1391  /* Decrypt the remaining bytes and return them in the payload buffer */
1392  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
1393  err = noise_symmetricstate_decrypt_and_hash(state->symmetric, &msg);
1394  if (err != NOISE_ERROR_NONE)
1395  return err;
1396  if (payload) {
1397  if (msg.size > payload->max_size)
1398  return NOISE_ERROR_INVALID_LENGTH;
1399  memcpy(payload->data, msg.data, msg.size);
1400  payload->size = msg.size;
1401  }
1402  return NOISE_ERROR_NONE;
1403 }
