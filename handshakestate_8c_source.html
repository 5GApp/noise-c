70 static int noise_handshakestate_new
71  (NoiseHandshakeState **state, NoiseSymmetricState *symmetric, int role)
72 {
73  const uint8_t *pattern;
74  int requirements;
75  int dh_id;
76  uint8_t flags;
77  int err;
78 
79  /* Locate the information for the current handshake pattern */
80  pattern = noise_pattern_lookup(symmetric->id.pattern_id);
81  if (!pattern) {
82  noise_symmetricstate_free(symmetric);
83  return NOISE_ERROR_UNKNOWN_ID;
84  }
85  flags = pattern[0];
86  if (role == NOISE_ROLE_RESPONDER) {
87  /* Reverse the pattern flags so that the responder is "local" */
88  flags = noise_pattern_reverse_flags(flags);
89  }
90 
91  /* Create the HandshakeState object */
92  *state = noise_new(NoiseHandshakeState);
93  if (!(*state)) {
94  noise_symmetricstate_free(symmetric);
95  return NOISE_ERROR_NO_MEMORY;
96  }
97 
98  /* What keys do we require to be able to start the protocol? */
99  requirements = 0;
100  if (flags & (NOISE_PAT_FLAG_LOCAL_STATIC | NOISE_PAT_FLAG_LOCAL_REQUIRED))
101  requirements |= NOISE_REQ_LOCAL_REQUIRED;
102  if (flags & NOISE_PAT_FLAG_REMOTE_REQUIRED)
103  requirements |= NOISE_REQ_REMOTE_REQUIRED;
104  if (symmetric->id.prefix_id == NOISE_PREFIX_PSK)
105  requirements |= NOISE_REQ_PSK;
106  requirements |= NOISE_REQ_PROLOGUE;
107 
108  /* Initialize the HandshakeState */
109  (*state)->requirements = requirements;
110  (*state)->action = NOISE_ACTION_NONE;
111  (*state)->tokens = pattern + 1;
112  (*state)->role = role;
113  (*state)->symmetric = symmetric;
114 
115  /* Create DHState objects for all of the keys we will need later */
116  err = NOISE_ERROR_NONE;
117  dh_id = symmetric->id.dh_id;
118  if ((flags & NOISE_PAT_FLAG_LOCAL_STATIC) != 0)
119  err = noise_dhstate_new_by_id(&((*state)->dh_local_static), dh_id);
120  if ((flags & NOISE_PAT_FLAG_LOCAL_EMPEMERAL) != 0 && err == NOISE_ERROR_NONE)
121  err = noise_dhstate_new_by_id(&((*state)->dh_local_ephemeral), dh_id);
122  if ((flags & NOISE_PAT_FLAG_REMOTE_STATIC) != 0 && err == NOISE_ERROR_NONE)
123  err = noise_dhstate_new_by_id(&((*state)->dh_remote_static), dh_id);
124  if ((flags & NOISE_PAT_FLAG_REMOTE_EMPEMERAL) != 0 && err == NOISE_ERROR_NONE)
125  err = noise_dhstate_new_by_id(&((*state)->dh_remote_ephemeral), dh_id);
126 
127  /* Bail out if we had an error trying to create the DHState objects */
128  if (err != NOISE_ERROR_NONE) {
129  noise_handshakestate_free(*state);
130  return err;
131  }
132 
133  /* Ready to go */
134  return NOISE_ERROR_NONE;
135 }

359 int noise_handshakestate_set_pre_shared_key
360  (NoiseHandshakeState *state, const uint8_t *key, size_t key_len)
361 {
362  uint8_t temp[NOISE_MAX_HASHLEN];
363  NoiseHashState *hash;
364 
365  /* Validate the parameters and state */
366  if (!state || !key)
367  return NOISE_ERROR_INVALID_PARAM;
368  if (state->symmetric->id.prefix_id != NOISE_PREFIX_PSK)
369  return NOISE_ERROR_NOT_APPLICABLE;
370  if (state->action != NOISE_ACTION_NONE)
371  return NOISE_ERROR_INVALID_STATE;
372  if (!(state->requirements & NOISE_REQ_PSK))
373  return NOISE_ERROR_INVALID_STATE;
374 
375  /* If we haven't hashed the prologue yet, hash an empty one now */
376  if (state->requirements & NOISE_REQ_PROLOGUE)
377  noise_handshakestate_set_prologue(state, "", 0);
378 
379  /* Mix the pre shared key into the chaining key and handshake hash */
380  hash = state->symmetric->hash;
381  noise_hashstate_hkdf
382  (hash, state->symmetric->ck, hash->hash_len, key, key_len,
383  state->symmetric->ck, hash->hash_len, temp, hash->hash_len);
384  noise_symmetricstate_mix_hash(state->symmetric, temp, hash->hash_len);
385  noise_clean(temp, sizeof(temp));
386 
387  /* We have the pre shared key now */
388  state->requirements &= ~NOISE_REQ_PSK;
389  return NOISE_ERROR_NONE;
390 }

567 int noise_handshakestate_start(NoiseHandshakeState *state)
568 {
569  /* Validate the parameter */
570  if (!state)
571  return NOISE_ERROR_INVALID_PARAM;
572  if (state->action != NOISE_ACTION_NONE)
573  return NOISE_ERROR_INVALID_STATE;
574  if (state->symmetric->id.pattern_id == NOISE_PATTERN_XX_FALLBACK &&
575  (state->requirements & NOISE_REQ_FALLBACK_EPHEM) == 0)
576  return NOISE_ERROR_NOT_APPLICABLE;
577 
578  /* Check that we have satisfied all of the pattern requirements */
579  if ((state->requirements & NOISE_REQ_LOCAL_REQUIRED) != 0 &&
580  !noise_dhstate_has_keypair(state->dh_local_static))
581  return NOISE_ERROR_LOCAL_KEY_REQUIRED;
582  if ((state->requirements & NOISE_REQ_REMOTE_REQUIRED) != 0 &&
583  !noise_dhstate_has_public_key(state->dh_remote_static))
584  return NOISE_ERROR_REMOTE_KEY_REQUIRED;
585  if ((state->requirements & NOISE_REQ_PSK) != 0)
586  return NOISE_ERROR_PSK_REQUIRED;
587 
588  /* If the prologue has not been provided yet, hash an empty one */
589  if (state->requirements & NOISE_REQ_PROLOGUE)
590  noise_handshakestate_set_prologue(state, "", 0);
591 
592  /* Mix the pre-supplied public keys into the handshake hash */
593  if (state->role == NOISE_ROLE_INITIATOR) {
594  noise_handshakestate_mix_public_key(state, state->dh_local_static);
595  noise_handshakestate_mix_public_key(state, state->dh_local_ephemeral);
596  noise_handshakestate_mix_public_key(state, state->dh_remote_static);
597  noise_handshakestate_mix_public_key(state, state->dh_remote_ephemeral);
598  } else {
599  noise_handshakestate_mix_public_key(state, state->dh_remote_static);
600  noise_handshakestate_mix_public_key(state, state->dh_remote_ephemeral);
601  noise_handshakestate_mix_public_key(state, state->dh_local_static);
602  noise_handshakestate_mix_public_key(state, state->dh_local_ephemeral);
603  }
604 
605  /* The handshake has now officially started */
606  if (state->role == NOISE_ROLE_INITIATOR)
607  state->action = NOISE_ACTION_WRITE_MESSAGE;
608  else
609  state->action = NOISE_ACTION_READ_MESSAGE;
610  return NOISE_ERROR_NONE;
611 }

645 int noise_handshakestate_fallback(NoiseHandshakeState *state)
646 {
647  char name[NOISE_MAX_PROTOCOL_NAME];
648  size_t hash_len;
649  size_t name_len;
650  NoiseProtocolId id;
651  int err;
652 
653  /* Validate the parameter */
654  if (!state)
655  return NOISE_ERROR_INVALID_PARAM;
656  if (state->symmetric->id.pattern_id != NOISE_PATTERN_IK)
657  return NOISE_ERROR_NOT_APPLICABLE;
658 
659  /* The initiator should be waiting for a return message from the
660  responder, and the responder should have failed on the first
661  handshake message from the initiator. We also allow the
662  responder to fallback after processing the first message
663  successfully; it decides to always fall back anyway. */
664  if (state->role == NOISE_ROLE_INITIATOR) {
665  if (state->action != NOISE_ACTION_READ_MESSAGE)
666  return NOISE_ERROR_INVALID_STATE;
667  if (!noise_dhstate_has_public_key(state->dh_local_ephemeral))
668  return NOISE_ERROR_INVALID_STATE;
669  } else {
670  if (state->action != NOISE_ACTION_FAILED &&
671  state->action != NOISE_ACTION_WRITE_MESSAGE)
672  return NOISE_ERROR_INVALID_STATE;
673  if (!noise_dhstate_has_public_key(state->dh_remote_ephemeral))
674  return NOISE_ERROR_INVALID_STATE;
675  }
676 
677  /* Format a new protocol name for the "XXfallback" variant */
678  id = state->symmetric->id;
679  id.pattern_id = NOISE_PATTERN_XX_FALLBACK;
680  err = noise_protocol_id_to_name(name, sizeof(name), &id);
681  if (err != NOISE_ERROR_NONE)
682  return err;
683 
684  /* Convert the HandshakeState to the "XXfallback" pattern */
685  state->symmetric->id.pattern_id = NOISE_PATTERN_XX_FALLBACK;
686  noise_dhstate_clear_key(state->dh_remote_static);
687  if (state->role == NOISE_ROLE_INITIATOR) {
688  noise_dhstate_clear_key(state->dh_remote_ephemeral);
689  state->action = NOISE_ACTION_READ_MESSAGE;
690  state->role = NOISE_ROLE_RESPONDER;
691  } else {
692  noise_dhstate_clear_key(state->dh_local_ephemeral);
693  state->action = NOISE_ACTION_WRITE_MESSAGE;
694  state->role = NOISE_ROLE_INITIATOR;
695  }
696  state->requirements |= NOISE_REQ_FALLBACK_EPHEM;
697 
698  /* Re-initialize the chaining key "ck" and the handshake hash "h" from
699  the new protocol name. If the name is too long, hash it down first */
700  name_len = strlen(name);
701  hash_len = noise_hashstate_get_hash_length(state->symmetric->hash);
702  if (name_len <= hash_len) {
703  memcpy(state->symmetric->h, name, name_len);
704  memset(state->symmetric->h + name_len, 0, hash_len - name_len);
705  } else {
706  noise_hashstate_hash_one
707  (state->symmetric->hash, (const uint8_t *)name, name_len,
708  state->symmetric->h);
709  }
710  memcpy(state->symmetric->ck, state->symmetric->h, hash_len);
711 
712  /* Reset the encryption key within the symmetric state to empty */
713  state->symmetric->cipher->has_key = 0;
714  state->symmetric->cipher->n = 0;
715  state->symmetric->cipher->nonce_overflow = 0;
716 
717  /* Ready to go */
718  return NOISE_ERROR_NONE;
719 }

793 static int noise_handshakestate_write
794  (NoiseHandshakeState *state, const void *payload, size_t payload_size,
795  uint8_t *message, size_t *message_size, size_t max_size)
796 {
797  size_t size = 0;
798  size_t len;
799  size_t out_len;
800  size_t mac_len;
801  uint8_t token;
802  int err;
803 
804  /* Process tokens until the direction changes or the pattern ends */
805  for (;;) {
806  token = *(state->tokens);
807  if (token == NOISE_TOKEN_END) {
808  /* The pattern has finished, so the next action is "split" */
809  state->action = NOISE_ACTION_SPLIT;
810  break;
811  } else if (token == NOISE_TOKEN_FLIP_DIR) {
812  /* Changing directions, so this message is complete and
813  the next action is "read message". */
814  ++(state->tokens);
815  state->action = NOISE_ACTION_READ_MESSAGE;
816  break;
817  }
818  err = NOISE_ERROR_NONE;
819  switch (token) {
820  case NOISE_TOKEN_E:
821  /* Generate a local ephemeral keypair and add the public
822  key to the message */
823  if (!state->dh_local_ephemeral)
824  return NOISE_ERROR_INVALID_STATE;
825  len = state->dh_local_ephemeral->public_key_len;
826  err = noise_dhstate_generate_keypair(state->dh_local_ephemeral);
827  if (err != NOISE_ERROR_NONE)
828  break;
829  if ((max_size - size) < len)
830  return NOISE_ERROR_INVALID_LENGTH;
831  memcpy(message + size, state->dh_local_ephemeral->public_key, len);
832  noise_symmetricstate_mix_hash
833  (state->symmetric, message + size, len);
834  size += len;
835  break;
836  case NOISE_TOKEN_S:
837  /* Encrypt the local static public key and add it to the message */
838  if (!state->dh_local_static)
839  return NOISE_ERROR_INVALID_STATE;
840  len = state->dh_local_static->public_key_len;
841  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
842  if ((max_size - size) < (len + mac_len))
843  return NOISE_ERROR_INVALID_LENGTH;
844  memcpy(message + size, state->dh_local_static->public_key, len);
845  err = noise_symmetricstate_encrypt_and_hash
846  (state->symmetric, message + size, len, &out_len);
847  if (err != NOISE_ERROR_NONE)
848  break;
849  size += out_len;
850  break;
851  case NOISE_TOKEN_DHEE:
852  /* DH operation with local and remote ephemeral keys */
853  err = noise_handshake_mix_dh
854  (state, state->dh_local_ephemeral, state->dh_remote_ephemeral);
855  break;
856  case NOISE_TOKEN_DHES:
857  /* DH operation with local ephemeral and remote static keys */
858  err = noise_handshake_mix_dh
859  (state, state->dh_local_ephemeral, state->dh_remote_static);
860  break;
861  case NOISE_TOKEN_DHSE:
862  /* DH operation with local static and remote ephemeral keys */
863  err = noise_handshake_mix_dh
864  (state, state->dh_local_static, state->dh_remote_ephemeral);
865  break;
866  case NOISE_TOKEN_DHSS:
867  /* DH operation with local and remote static keys */
868  err = noise_handshake_mix_dh
869  (state, state->dh_local_static, state->dh_remote_static);
870  break;
871  default:
872  /* Unknown token code in the pattern. This shouldn't happen.
873  If it does, then abort immediately. */
874  err = NOISE_ERROR_INVALID_STATE;
875  break;
876  }
877  if (err != NOISE_ERROR_NONE)
878  return err;
879  ++(state->tokens);
880  }
881 
882  /* Encrypt the payload and add it to the buffer */
883  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
884  if ((max_size - size) < mac_len)
885  return NOISE_ERROR_INVALID_LENGTH;
886  if ((max_size - size - mac_len) < payload_size)
887  return NOISE_ERROR_INVALID_LENGTH;
888  if (payload_size)
889  memcpy(message + size, payload, payload_size);
890  err = noise_symmetricstate_encrypt_and_hash
891  (state->symmetric, message + size, payload_size, &out_len);
892  if (err != NOISE_ERROR_NONE)
893  return err;
894  size += out_len;
895 
896  /* Return the final size to the caller */
897  *message_size = size;
898  return NOISE_ERROR_NONE;
899 }

928 int noise_handshakestate_write_message
929  (NoiseHandshakeState *state, const void *payload, size_t payload_size,
930  uint8_t *message, size_t *message_size)
931 {
932  size_t max_size = 0;
933  int err = NOISE_ERROR_NONE;
934 
935  /* Validate the message size and extract it. We set the return size
936  to zero and clear the message buffer in case this function bails out
937  with an error later. This is in case the caller forgets to check the
938  error return value, and simply transmits the contents of "message".
939  This will prevent the caller from accidentally leaking the previous
940  contents of "message" onto the transport. */
941  if (message_size) {
942  max_size = *message_size;
943  *message_size = 0;
944  if (message)
945  noise_clean(message, max_size);
946  } else {
947  err = NOISE_ERROR_INVALID_PARAM;
948  }
949 
950  /* Validate the other parameters and state */
951  if (err != NOISE_ERROR_NONE) {
952  if (!state || !message)
953  err = NOISE_ERROR_INVALID_PARAM;
954  else if (!payload && payload_size != 0)
955  err = NOISE_ERROR_INVALID_PARAM;
956  else if (state->action != NOISE_ACTION_WRITE_MESSAGE)
957  err = NOISE_ERROR_INVALID_STATE;
958  }
959 
960  /* Perform the main write if no error so far */
961  if (err == NOISE_ERROR_NONE) {
962  err = noise_handshakestate_write
963  (state, payload, payload_size, message, message_size, max_size);
964  }
965 
966  /* If an error occurred, then fail the HandshakeState completely */
967  if (err != NOISE_ERROR_NONE && state) {
968  state->action = NOISE_ACTION_FAILED;
969  if (message) {
970  /* Clear the message buffer again in case we wrote some
971  partial data to it before discovering the error */
972  noise_clean(message, max_size);
973  }
974  }
975 
976  /* Finished */
977  return err;
978 }

995  (NoiseHandshakeState *state, uint8_t *message, size_t message_size,
996  void *payload, size_t *payload_size, size_t max_size)
997 {
998  size_t len;
999  size_t out_len;
1000  size_t mac_len;
1001  uint8_t token;
1002  int err;
1003 
1004  /* Process tokens until the direction changes or the pattern ends */
1005  for (;;) {
1006  token = *(state->tokens);
1007  if (token == NOISE_TOKEN_END) {
1008  /* The pattern has finished, so the next action is "split" */
1009  state->action = NOISE_ACTION_SPLIT;
1010  break;
1011  } else if (token == NOISE_TOKEN_FLIP_DIR) {
1012  /* Changing directions, so this message is complete and
1013  the next action is "read message". */
1014  ++(state->tokens);
1015  state->action = NOISE_ACTION_WRITE_MESSAGE;
1016  break;
1017  }
1018  err = NOISE_ERROR_NONE;
1019  switch (token) {
1020  case NOISE_TOKEN_E:
1021  /* Save the remote ephemeral key and hash it */
1022  if (!state->dh_remote_ephemeral)
1023  return NOISE_ERROR_INVALID_STATE;
1024  len = state->dh_remote_ephemeral->public_key_len;
1025  if (message_size < len)
1026  return NOISE_ERROR_INVALID_LENGTH;
1027  err = noise_dhstate_set_public_key
1028  (state->dh_remote_ephemeral, message, len);
1029  if (err != NOISE_ERROR_NONE)
1030  break;
1031  if (noise_dhstate_is_null_public_key(state->dh_remote_ephemeral)) {
1032  /* The remote ephemeral key is null, which means that it is
1033  not contributing anything to the security of the session
1034  and is in fact downgrading the security to "none at all"
1035  in some of the message patterns. Reject all such keys. */
1036  return NOISE_ERROR_INVALID_PUBLIC_KEY;
1037  }
1038  message += len;
1039  message_size -= len;
1040  break;
1041  case NOISE_TOKEN_S:
1042  /* Decrypt and read the remote static key */
1043  if (!state->dh_remote_static)
1044  return NOISE_ERROR_INVALID_STATE;
1045  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
1046  len = state->dh_remote_static->public_key_len + mac_len;
1047  if (message_size < len)
1048  return NOISE_ERROR_INVALID_LENGTH;
1049  err = noise_symmetricstate_decrypt_and_hash
1050  (state->symmetric, message, len, &out_len);
1051  if (err != NOISE_ERROR_NONE)
1052  break;
1053  err = noise_dhstate_set_public_key
1054  (state->dh_remote_static, message, out_len);
1055  if (err != NOISE_ERROR_NONE)
1056  break;
1057  message += len;
1058  message_size -= len;
1059  break;
1060  case NOISE_TOKEN_DHEE:
1061  /* DH operation with local and remote ephemeral keys */
1062  err = noise_handshake_mix_dh
1063  (state, state->dh_local_ephemeral, state->dh_remote_ephemeral);
1064  break;
1065  case NOISE_TOKEN_DHES:
1066  /* DH operation with remote ephemeral and local static keys */
1067  err = noise_handshake_mix_dh
1068  (state, state->dh_local_static, state->dh_remote_ephemeral);
1069  break;
1070  case NOISE_TOKEN_DHSE:
1071  /* DH operation with remote static and local ephemeral keys */
1072  err = noise_handshake_mix_dh
1073  (state, state->dh_local_ephemeral, state->dh_remote_static);
1074  break;
1075  case NOISE_TOKEN_DHSS:
1076  /* DH operation with local and remote static keys */
1077  err = noise_handshake_mix_dh
1078  (state, state->dh_local_static, state->dh_remote_static);
1079  break;
1080  default:
1081  /* Unknown token code in the pattern. This shouldn't happen.
1082  If it does, then abort immediately. */
1083  err = NOISE_ERROR_INVALID_STATE;
1084  break;
1085  }
1086  if (err != NOISE_ERROR_NONE)
1087  return err;
1088  ++(state->tokens);
1089  }
1090 
1091  /* Decrypt the payload and return it in the payload buffer */
1092  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
1093  err = noise_symmetricstate_decrypt_and_hash
1094  (state->symmetric, message, message_size, &len);
1095  if (err != NOISE_ERROR_NONE)
1096  return err;
1097  if (len > max_size)
1098  return NOISE_ERROR_INVALID_LENGTH;
1099  if (payload)
1100  memcpy(payload, message, len);
1101 
1102  /* Return the final payload size to the caller */
1103  *payload_size = len;
1104  return NOISE_ERROR_NONE;
1105 }

1145 int noise_handshakestate_read_message
1146  (NoiseHandshakeState *state, uint8_t *message, size_t message_size,
1147  void *payload, size_t *payload_size)
1148 {
1149  size_t max_size = 0;
1150  int err = NOISE_ERROR_NONE;
1151 
1152  /* Validate the payload size and extract it. We set the return size
1153  to zero and clear the payload buffer in case this function bails out
1154  with an error later. This is in case the caller forgets to check
1155  the error return value and simply passes the payload bytes up to the
1156  application. This could lead to "replay attack" scenarios where the
1157  previous payload contents can be re-fed into the application to
1158  repeat a previous command. */
1159  if (payload_size) {
1160  max_size = *payload_size;
1161  *payload_size = 0;
1162  if (payload)
1163  noise_clean(payload, max_size);
1164  } else {
1165  err = NOISE_ERROR_INVALID_PARAM;
1166  }
1167 
1168  /* Validate the other parameters and state */
1169  if (err == NOISE_ERROR_NONE) {
1170  if (!state || !message)
1171  err = NOISE_ERROR_INVALID_PARAM;
1172  else if (state->action != NOISE_ACTION_READ_MESSAGE)
1173  err = NOISE_ERROR_INVALID_STATE;
1174  }
1175 
1176  /* Perform the main read if no error so far */
1177  if (err == NOISE_ERROR_NONE) {
1178  err = noise_handshakestate_read
1179  (state, message, message_size, payload, payload_size, max_size);
1180  }
1181 
1182  /* Clear the incoming message buffer to prevent leakage of decrypted data */
1183  if (message)
1184  noise_clean(message, message_size);
1185 
1186  /* If an error occurred, then fail the HandshakeState completely */
1187  if (err != NOISE_ERROR_NONE && state)
1188  state->action = NOISE_ACTION_FAILED;
1189 
1190  /* Finished */
1191  return err;
1192 }
