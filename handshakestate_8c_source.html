65 static int noise_handshakestate_new
66  (NoiseHandshakeState **state, NoiseSymmetricState *symmetric, int role)
67 {
68  const uint8_t *pattern;
69  NoiseDHState *dh;
70  int requirements;
71  uint8_t flags;
72  uint8_t *ptr;
73  size_t size;
74  int err;
75 
76  /* Locate the information for the current handshake pattern */
77  pattern = noise_pattern_lookup(symmetric->id.pattern_id);
78  if (!pattern) {
79  noise_symmetricstate_free(symmetric);
80  return NOISE_ERROR_UNKNOWN_ID;
81  }
82  flags = pattern[0];
83  if (role == NOISE_ROLE_RESPONDER) {
84  /* Reverse the pattern flags so that the responder is "local" */
85  flags = noise_pattern_reverse_flags(flags);
86  }
87 
88  /* Create the DHState object */
89  err = noise_dhstate_new_by_id(&dh, symmetric->id.dh_id);
90  if (err != NOISE_ERROR_NONE) {
91  noise_symmetricstate_free(symmetric);
92  return err;
93  }
94 
95  /* Determine how much memory we need for the HandshakeState,
96  plus all of the private/public key values we'll need later. */
97  size = sizeof(NoiseHandshakeState);
98  if (flags & NOISE_PAT_FLAG_LOCAL_STATIC)
99  size += dh->private_key_len + dh->public_key_len;
100  if (flags & NOISE_PAT_FLAG_LOCAL_EMPEMERAL)
101  size += dh->private_key_len + dh->public_key_len;
102  if (flags & NOISE_PAT_FLAG_REMOTE_STATIC)
103  size += dh->public_key_len;
104  if (flags & NOISE_PAT_FLAG_REMOTE_EMPEMERAL)
105  size += dh->public_key_len;
106 
107  /* Allocate the HandshakeState */
108  *state = noise_new_object(size);
109  if (!(*state)) {
110  noise_symmetricstate_free(symmetric);
111  noise_dhstate_free(dh);
112  return NOISE_ERROR_NO_MEMORY;
113  }
114 
115  /* What keys do we require to be able to start the protocol? */
116  requirements = 0;
117  if (flags & (NOISE_PAT_FLAG_LOCAL_STATIC | NOISE_PAT_FLAG_LOCAL_REQUIRED))
118  requirements |= NOISE_REQ_LOCAL_REQUIRED;
119  if (flags & NOISE_PAT_FLAG_REMOTE_REQUIRED)
120  requirements |= NOISE_REQ_REMOTE_REQUIRED;
121  if (symmetric->id.prefix_id == NOISE_PREFIX_PSK)
122  requirements |= NOISE_REQ_PSK;
123 
124  /* Initialize the HandshakeState */
125  (*state)->requirements = requirements;
126  (*state)->tokens = pattern + 1;
127  (*state)->role = role;
128  (*state)->symmetric = symmetric;
129  (*state)->dh = dh;
130 
131  /* Set the pointers for all of the key values we'll need later */
132  #define alloc_key(name, type) \
133  ((*state)->name##_##type = ptr, ptr += dh->type##_len)
134  ptr = ((uint8_t *)(*state)) + sizeof(NoiseHandshakeState);
135  if (flags & NOISE_PAT_FLAG_LOCAL_STATIC) {
136  alloc_key(local_static, private_key);
137  alloc_key(local_static, public_key);
138  }
139  if (flags & NOISE_PAT_FLAG_LOCAL_EMPEMERAL) {
140  alloc_key(local_ephemeral, private_key);
141  alloc_key(local_ephemeral, public_key);
142  }
143  if (flags & NOISE_PAT_FLAG_REMOTE_STATIC) {
144  alloc_key(remote_static, public_key);
145  }
146  if (flags & NOISE_PAT_FLAG_REMOTE_EMPEMERAL) {
147  alloc_key(remote_ephemeral, public_key);
148  }
149 
150  /* Ready to go */
151  return NOISE_ERROR_NONE;
152 }

179 {

259 int noise_handshakestate_free(NoiseHandshakeState *state)
260 {
261  /* Bail out if no handshake state */
262  if (!state)
263  return NOISE_ERROR_INVALID_PARAM;
264 
265  /* Free the sub objects that are hanging off the handshakestate */
266  if (state->symmetric)
267  noise_symmetricstate_free(state->symmetric);
268  if (state->dh)
269  noise_dhstate_free(state->dh);
270 
271  /* Clean and free the memory for "state" */
272  noise_free(state, state->size);
273  return NOISE_ERROR_NONE;
274 }
