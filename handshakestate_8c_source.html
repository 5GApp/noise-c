60 static int noise_handshakestate_requirements
61  (uint8_t flags, int prefix_id, int role, int is_fallback)
62 {
63  int requirements = 0;
64  if (flags & NOISE_PAT_FLAG_LOCAL_STATIC) {
65  requirements |= NOISE_REQ_LOCAL_REQUIRED;
66  }
67  if (flags & NOISE_PAT_FLAG_LOCAL_REQUIRED) {
68  requirements |= NOISE_REQ_LOCAL_REQUIRED;
69  requirements |= NOISE_REQ_LOCAL_PREMSG;
70  }
71  if (flags & NOISE_PAT_FLAG_REMOTE_REQUIRED) {
72  requirements |= NOISE_REQ_REMOTE_REQUIRED;
73  requirements |= NOISE_REQ_REMOTE_PREMSG;
74  }
75  if (flags & (NOISE_PAT_FLAG_REMOTE_EPHEM_REQ |
76  NOISE_PAT_FLAG_LOCAL_EPHEM_REQ)) {
77  if (is_fallback)
78  requirements |= NOISE_REQ_FALLBACK_PREMSG;
79  }
80  if (prefix_id == NOISE_PREFIX_PSK) {
81  requirements |= NOISE_REQ_PSK;
82  }
83  return requirements;
84 }

102 static int noise_handshakestate_new
103  (NoiseHandshakeState **state, NoiseSymmetricState *symmetric, int role)
104 {
105  const uint8_t *pattern;
106  int dh_id;
107  uint8_t flags;
108  int extra_reqs = 0;
109  int err;
110 
111  /* Locate the information for the current handshake pattern */
112  pattern = noise_pattern_lookup(symmetric->id.pattern_id);
113  if (!pattern) {
114  noise_symmetricstate_free(symmetric);
115  return NOISE_ERROR_UNKNOWN_ID;
116  }
117  flags = pattern[0];
118  if ((flags & NOISE_PAT_FLAG_REMOTE_REQUIRED) != 0)
119  extra_reqs |= NOISE_REQ_FALLBACK_POSSIBLE;
120  if (role == NOISE_ROLE_RESPONDER) {
121  /* Reverse the pattern flags so that the responder is "local" */
122  flags = noise_pattern_reverse_flags(flags);
123  }
124 
125  /* Create the HandshakeState object */
126  *state = noise_new(NoiseHandshakeState);
127  if (!(*state)) {
128  noise_symmetricstate_free(symmetric);
129  return NOISE_ERROR_NO_MEMORY;
130  }
131 
132  /* Initialize the HandshakeState */
133  (*state)->requirements = extra_reqs | noise_handshakestate_requirements
134  (flags, symmetric->id.prefix_id, role, 0);
135  (*state)->action = NOISE_ACTION_NONE;
136  (*state)->tokens = pattern + 1;
137  (*state)->role = role;
138  (*state)->symmetric = symmetric;
139 
140  /* Create DHState objects for all of the keys we will need later */
141  err = NOISE_ERROR_NONE;
142  dh_id = symmetric->id.dh_id;
143  if ((flags & NOISE_PAT_FLAG_LOCAL_STATIC) != 0)
144  err = noise_dhstate_new_by_id(&((*state)->dh_local_static), dh_id);
145  if ((flags & NOISE_PAT_FLAG_LOCAL_EPHEMERAL) != 0 && err == NOISE_ERROR_NONE)
146  err = noise_dhstate_new_by_id(&((*state)->dh_local_ephemeral), dh_id);
147  if ((flags & NOISE_PAT_FLAG_REMOTE_STATIC) != 0 && err == NOISE_ERROR_NONE)
148  err = noise_dhstate_new_by_id(&((*state)->dh_remote_static), dh_id);
149  if ((flags & NOISE_PAT_FLAG_REMOTE_EPHEMERAL) != 0 && err == NOISE_ERROR_NONE)
150  err = noise_dhstate_new_by_id(&((*state)->dh_remote_ephemeral), dh_id);
151 
152  /* Bail out if we had an error trying to create the DHState objects */
153  if (err != NOISE_ERROR_NONE) {
154  noise_handshakestate_free(*state);
155  return err;
156  }
157 
158  /* Ready to go */
159  return NOISE_ERROR_NONE;
160 }

678 int noise_handshakestate_start(NoiseHandshakeState *state)
679 {
680  /* Validate the parameter */
681  if (!state)
682  return NOISE_ERROR_INVALID_PARAM;
683  if (state->action != NOISE_ACTION_NONE)
684  return NOISE_ERROR_INVALID_STATE;
685  if (state->symmetric->id.pattern_id == NOISE_PATTERN_XX_FALLBACK &&
686  (state->requirements & NOISE_REQ_FALLBACK_PREMSG) == 0)
687  return NOISE_ERROR_NOT_APPLICABLE;
688 
689  /* Check that we have satisfied all of the pattern requirements */
690  if ((state->requirements & NOISE_REQ_LOCAL_REQUIRED) != 0 &&
691  !noise_dhstate_has_keypair(state->dh_local_static))
692  return NOISE_ERROR_LOCAL_KEY_REQUIRED;
693  if ((state->requirements & NOISE_REQ_REMOTE_REQUIRED) != 0 &&
694  !noise_dhstate_has_public_key(state->dh_remote_static))
695  return NOISE_ERROR_REMOTE_KEY_REQUIRED;
696  if ((state->requirements & NOISE_REQ_PSK) != 0 &&
697  state->pre_shared_key_len == 0)
698  return NOISE_ERROR_PSK_REQUIRED;
699 
700  /* Hash the prologue value */
701  if (state->prologue_len) {
702  noise_symmetricstate_mix_hash
703  (state->symmetric, state->prologue, state->prologue_len);
704  } else {
705  /* No prologue, so hash an empty one */
706  noise_symmetricstate_mix_hash
707  (state->symmetric, state->pre_shared_key, 0);
708  }
709 
710  /* Mix the pre shared key into the chaining key and handshake hash */
711  if (state->pre_shared_key_len) {
712  uint8_t temp[NOISE_MAX_HASHLEN];
713  NoiseHashState *hash = state->symmetric->hash;
714  noise_hashstate_hkdf
715  (hash, state->symmetric->ck, hash->hash_len,
716  state->pre_shared_key, state->pre_shared_key_len,
717  state->symmetric->ck, hash->hash_len, temp, hash->hash_len);
718  noise_symmetricstate_mix_hash(state->symmetric, temp, hash->hash_len);
719  noise_clean(temp, sizeof(temp));
720  }
721 
722  /* Mix the pre-supplied public keys into the handshake hash */
723  if (state->role == NOISE_ROLE_INITIATOR) {
724  if (state->requirements & NOISE_REQ_LOCAL_PREMSG)
725  noise_handshakestate_mix_public_key(state, state->dh_local_static);
726  if (state->requirements & NOISE_REQ_FALLBACK_PREMSG) {
727  noise_handshakestate_mix_public_key(state, state->dh_remote_ephemeral);
728  if ((state->requirements & NOISE_REQ_PSK) != 0) {
729  noise_handshakestate_mix_chaining_key
730  (state, state->dh_remote_ephemeral);
731  }
732  }
733  if (state->requirements & NOISE_REQ_REMOTE_PREMSG)
734  noise_handshakestate_mix_public_key(state, state->dh_remote_static);
735  } else {
736  if (state->requirements & NOISE_REQ_REMOTE_PREMSG)
737  noise_handshakestate_mix_public_key(state, state->dh_remote_static);
738  if (state->requirements & NOISE_REQ_FALLBACK_PREMSG) {
739  noise_handshakestate_mix_public_key(state, state->dh_local_ephemeral);
740  if ((state->requirements & NOISE_REQ_PSK) != 0) {
741  noise_handshakestate_mix_chaining_key
742  (state, state->dh_local_ephemeral);
743  }
744  }
745  if (state->requirements & NOISE_REQ_LOCAL_PREMSG)
746  noise_handshakestate_mix_public_key(state, state->dh_local_static);
747  }
748 
749  /* The handshake has now officially started */
750  if (state->role == NOISE_ROLE_INITIATOR)
751  state->action = NOISE_ACTION_WRITE_MESSAGE;
752  else
753  state->action = NOISE_ACTION_READ_MESSAGE;
754  return NOISE_ERROR_NONE;
755 }

843 int noise_handshakestate_fallback_to(NoiseHandshakeState *state, int pattern_id)
844 {
845  char name[NOISE_MAX_PROTOCOL_NAME];
846  size_t hash_len;
847  size_t name_len;
848  NoiseProtocolId id;
849  const uint8_t *pattern;
850  uint8_t flags;
851  int err;
852 
853  /* Validate the parameter */
854  if (!state)
855  return NOISE_ERROR_INVALID_PARAM;
856 
857  /* The original pattern must end in "K" for fallback to be possible */
858  if (state->symmetric->id.pattern_id < NOISE_PATTERN_NN ||
859  (state->requirements & NOISE_REQ_FALLBACK_POSSIBLE) == 0)
860  return NOISE_ERROR_NOT_APPLICABLE;
861 
862  /* Check that "pattern_id" supports fallback */
863  pattern = noise_pattern_lookup(pattern_id);
864  if (!pattern || (*pattern & NOISE_PAT_FLAG_REMOTE_EPHEM_REQ) == 0)
865  return NOISE_ERROR_NOT_APPLICABLE;
866 
867  /* The initiator should be waiting for a return message from the
868  responder, and the responder should have failed on the first
869  handshake message from the initiator. We also allow the
870  responder to fallback after processing the first message
871  successfully; it decides to always fall back anyway. */
872  if (state->role == NOISE_ROLE_INITIATOR) {
873  if (state->action != NOISE_ACTION_FAILED &&
874  state->action != NOISE_ACTION_READ_MESSAGE)
875  return NOISE_ERROR_INVALID_STATE;
876  if (!noise_dhstate_has_public_key(state->dh_local_ephemeral))
877  return NOISE_ERROR_INVALID_STATE;
878  } else {
879  if (state->action != NOISE_ACTION_FAILED &&
880  state->action != NOISE_ACTION_WRITE_MESSAGE)
881  return NOISE_ERROR_INVALID_STATE;
882  if (!noise_dhstate_has_public_key(state->dh_remote_ephemeral))
883  return NOISE_ERROR_INVALID_STATE;
884  }
885 
886  /* Format a new protocol name for the fallback variant */
887  id = state->symmetric->id;
888  id.pattern_id = pattern_id;
889  err = noise_protocol_id_to_name(name, sizeof(name), &id);
890  if (err != NOISE_ERROR_NONE)
891  return err;
892 
893  /* Convert the HandshakeState to the fallback pattern */
894  state->symmetric->id.pattern_id = pattern_id;
895  if (state->role == NOISE_ROLE_INITIATOR) {
896  noise_dhstate_clear_key(state->dh_remote_ephemeral);
897  noise_dhstate_clear_key(state->dh_remote_static);
898  state->role = NOISE_ROLE_RESPONDER;
899  } else {
900  noise_dhstate_clear_key(state->dh_local_ephemeral);
901  if (!(pattern[0] & NOISE_PAT_FLAG_REMOTE_REQUIRED))
902  noise_dhstate_clear_key(state->dh_remote_static);
903  state->role = NOISE_ROLE_INITIATOR;
904  }
905 
906  /* Start a new token pattern for the fallback */
907  state->tokens = pattern + 1;
908  state->action = NOISE_ACTION_NONE;
909 
910  /* Set up the key requirements for the fallback */
911  flags = pattern[0];
912  if (state->role == NOISE_ROLE_RESPONDER) {
913  flags = noise_pattern_reverse_flags(flags);
914  }
915  state->requirements = noise_handshakestate_requirements
916  (flags, id.prefix_id, state->role, 1);
917 
918  /* Re-initialize the chaining key "ck" and the handshake hash "h" from
919  the new protocol name. If the name is too long, hash it down first */
920  name_len = strlen(name);
921  hash_len = noise_hashstate_get_hash_length(state->symmetric->hash);
922  if (name_len <= hash_len) {
923  memcpy(state->symmetric->h, name, name_len);
924  memset(state->symmetric->h + name_len, 0, hash_len - name_len);
925  } else {
926  noise_hashstate_hash_one
927  (state->symmetric->hash, (const uint8_t *)name, name_len,
928  state->symmetric->h, hash_len);
929  }
930  memcpy(state->symmetric->ck, state->symmetric->h, hash_len);
931 
932  /* Reset the encryption key within the symmetric state to empty */
933  state->symmetric->cipher->has_key = 0;
934  state->symmetric->cipher->n = 0;
935 
936  /* Ready to go */
937  return NOISE_ERROR_NONE;
938 }

1010 static int noise_handshakestate_write
1011  (NoiseHandshakeState *state, NoiseBuffer *message, const NoiseBuffer *payload)
1012 {
1013  NoiseBuffer rest;
1014  size_t len;
1015  size_t mac_len;
1016  uint8_t token;
1017  int err;
1018 
1019  /* Process tokens until the direction changes or the pattern ends */
1020  for (;;) {
1021  token = *(state->tokens);
1022  if (token == NOISE_TOKEN_END) {
1023  /* The pattern has finished, so the next action is "split" */
1024  state->action = NOISE_ACTION_SPLIT;
1025  break;
1026  } else if (token == NOISE_TOKEN_FLIP_DIR) {
1027  /* Changing directions, so this message is complete and
1028  the next action is "read message". */
1029  ++(state->tokens);
1030  state->action = NOISE_ACTION_READ_MESSAGE;
1031  break;
1032  }
1033 
1034  /* Set "rest" to the rest of the "message" buffer after the
1035  current size. This is the space we have left to write
1036  handshake values while processing this token. */
1037  rest.data = message->data + message->size;
1038  rest.size = 0;
1039  rest.max_size = message->max_size - message->size;
1040 
1041  /* Process the token */
1042  err = NOISE_ERROR_NONE;
1043  switch (token) {
1044  case NOISE_TOKEN_E:
1045  /* Generate a local ephemeral keypair and add the public
1046  key to the message. If we are running fixed vector tests,
1047  then the ephemeral key may have already been provided. */
1048  if (!state->dh_local_ephemeral)
1049  return NOISE_ERROR_INVALID_STATE;

1225 static int noise_handshakestate_read
1226  (NoiseHandshakeState *state, NoiseBuffer *message, NoiseBuffer *payload)
1227 {
1228  NoiseBuffer msg;
1229  NoiseBuffer msg2;
1230  size_t len;
1231  size_t mac_len;
1232  uint8_t token;
1233  int err;
1234 
1235  /* Make a copy of the message buffer. As we process tokens, the copy
1236  will become shorter and shorter until only the payload is left. */
1237  msg = *message;
1238 
1239  /* Process tokens until the direction changes or the pattern ends */
1240  for (;;) {
1241  token = *(state->tokens);
1242  if (token == NOISE_TOKEN_END) {
1243  /* The pattern has finished, so the next action is "split" */
1244  state->action = NOISE_ACTION_SPLIT;
1245  break;
1246  } else if (token == NOISE_TOKEN_FLIP_DIR) {
1247  /* Changing directions, so this message is complete and
1248  the next action is "read message". */
1249  ++(state->tokens);
1250  state->action = NOISE_ACTION_WRITE_MESSAGE;
1251  break;
1252  }
1253  err = NOISE_ERROR_NONE;
1254  switch (token) {
1255  case NOISE_TOKEN_E:
1256  /* Save the remote ephemeral key and hash it */
1257  if (!state->dh_remote_ephemeral)
1258  return NOISE_ERROR_INVALID_STATE;
1259  len = state->dh_remote_ephemeral->public_key_len;
1260  if (msg.size < len)
1261  return NOISE_ERROR_INVALID_LENGTH;
1262  err = noise_symmetricstate_mix_hash
1263  (state->symmetric, msg.data, len);
1264  if (err != NOISE_ERROR_NONE)
1265  break;
1266  err = noise_dhstate_set_public_key
1267  (state->dh_remote_ephemeral, msg.data, len);
1268  if (err != NOISE_ERROR_NONE)
1269  break;
1270  if (noise_dhstate_is_null_public_key(state->dh_remote_ephemeral)) {
1271  /* The remote ephemeral key is null, which means that it is
1272  not contributing anything to the security of the session
1273  and is in fact downgrading the security to "none at all"
1274  in some of the message patterns. Reject all such keys. */
1275  return NOISE_ERROR_INVALID_PUBLIC_KEY;
1276  }
1277  msg.data += len;
1278  msg.size -= len;
1279  msg.max_size -= len;
1280 
1281  /* If the protocol is using pre-shared keys, then also mix
1282  the remote ephemeral key into the chaining key */
1283  if (state->symmetric->id.prefix_id == NOISE_PREFIX_PSK) {
1284  err = noise_symmetricstate_mix_key
1285  (state->symmetric,
1286  state->dh_remote_ephemeral->public_key, len);
1287  }
1288  break;
1289  case NOISE_TOKEN_S:
1290  /* Decrypt and read the remote static key */
1291  if (!state->dh_remote_static)
1292  return NOISE_ERROR_INVALID_STATE;
1293  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
1294  len = state->dh_remote_static->public_key_len + mac_len;
1295  if (msg.size < len)
1296  return NOISE_ERROR_INVALID_LENGTH;
1297  msg2.data = msg.data;
1298  msg2.size = len;
1299  msg2.max_size = len;
1300  err = noise_symmetricstate_decrypt_and_hash
1301  (state->symmetric, &msg2);
1302  if (err != NOISE_ERROR_NONE)
1303  break;
1304  err = noise_dhstate_set_public_key
1305  (state->dh_remote_static, msg2.data, msg2.size);
1306  if (err != NOISE_ERROR_NONE)
1307  break;
1308  msg.data += len;
1309  msg.size -= len;
1310  msg.max_size -= len;
1311  break;
1312  case NOISE_TOKEN_DHEE:
1313  /* DH operation with local and remote ephemeral keys */
1314  err = noise_handshake_mix_dh
1315  (state, state->dh_local_ephemeral, state->dh_remote_ephemeral);
1316  break;
1317  case NOISE_TOKEN_DHES:
1318  /* DH operation with remote ephemeral and local static keys */
1319  err = noise_handshake_mix_dh
1320  (state, state->dh_local_static, state->dh_remote_ephemeral);
1321  break;
1322  case NOISE_TOKEN_DHSE:
1323  /* DH operation with remote static and local ephemeral keys */
1324  err = noise_handshake_mix_dh
1325  (state, state->dh_local_ephemeral, state->dh_remote_static);
1326  break;
1327  case NOISE_TOKEN_DHSS:
1328  /* DH operation with local and remote static keys */
1329  err = noise_handshake_mix_dh
1330  (state, state->dh_local_static, state->dh_remote_static);
1331  break;
1332  default:
1333  /* Unknown token code in the pattern. This shouldn't happen.
1334  If it does, then abort immediately. */
1335  err = NOISE_ERROR_INVALID_STATE;
1336  break;
1337  }
1338  if (err != NOISE_ERROR_NONE)
1339  return err;
1340  ++(state->tokens);
1341  }
1342 
1343  /* Decrypt the remaining bytes and return them in the payload buffer */
1344  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
1345  err = noise_symmetricstate_decrypt_and_hash(state->symmetric, &msg);
1346  if (err != NOISE_ERROR_NONE)
1347  return err;
1348  if (payload) {
1349  if (msg.size > payload->max_size)
1350  return NOISE_ERROR_INVALID_LENGTH;
1351  memcpy(payload->data, msg.data, msg.size);
1352  payload->size = msg.size;
1353  }
1354  return NOISE_ERROR_NONE;
1355 }
