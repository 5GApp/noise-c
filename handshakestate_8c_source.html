70 static int noise_handshakestate_new
71  (NoiseHandshakeState **state, NoiseSymmetricState *symmetric, int role)
72 {
73  const uint8_t *pattern;
74  int requirements;
75  int dh_id;
76  uint8_t flags;
77  int err;
78 
79  /* Locate the information for the current handshake pattern */
80  pattern = noise_pattern_lookup(symmetric->id.pattern_id);
81  if (!pattern) {
82  noise_symmetricstate_free(symmetric);
83  return NOISE_ERROR_UNKNOWN_ID;
84  }
85  flags = pattern[0];
86  if (role == NOISE_ROLE_RESPONDER) {
87  /* Reverse the pattern flags so that the responder is "local" */
88  flags = noise_pattern_reverse_flags(flags);
89  }
90 
91  /* Create the HandshakeState object */
92  *state = noise_new(NoiseHandshakeState);
93  if (!(*state)) {
94  noise_symmetricstate_free(symmetric);
95  return NOISE_ERROR_NO_MEMORY;
96  }
97 
98  /* What keys do we require to be able to start the protocol? */
99  requirements = 0;
100  if (flags & NOISE_PAT_FLAG_LOCAL_STATIC) {
101  requirements |= NOISE_REQ_LOCAL_REQUIRED;
102  } if (flags & NOISE_PAT_FLAG_LOCAL_REQUIRED) {
103  requirements |= NOISE_REQ_LOCAL_REQUIRED;
104  requirements |= NOISE_REQ_LOCAL_PREMSG;
105  }
106  if (flags & NOISE_PAT_FLAG_REMOTE_REQUIRED) {
107  requirements |= NOISE_REQ_REMOTE_REQUIRED;
108  requirements |= NOISE_REQ_REMOTE_PREMSG;
109  }
110  if (symmetric->id.prefix_id == NOISE_PREFIX_PSK) {
111  requirements |= NOISE_REQ_PSK;
112  }
113  requirements |= NOISE_REQ_PROLOGUE;
114 
115  /* Initialize the HandshakeState */
116  (*state)->requirements = requirements;
117  (*state)->action = NOISE_ACTION_NONE;
118  (*state)->tokens = pattern + 1;
119  (*state)->role = role;
120  (*state)->symmetric = symmetric;
121 
122  /* Create DHState objects for all of the keys we will need later */
123  err = NOISE_ERROR_NONE;
124  dh_id = symmetric->id.dh_id;
125  if ((flags & NOISE_PAT_FLAG_LOCAL_STATIC) != 0)
126  err = noise_dhstate_new_by_id(&((*state)->dh_local_static), dh_id);
127  if ((flags & NOISE_PAT_FLAG_LOCAL_EMPEMERAL) != 0 && err == NOISE_ERROR_NONE)
128  err = noise_dhstate_new_by_id(&((*state)->dh_local_ephemeral), dh_id);
129  if ((flags & NOISE_PAT_FLAG_REMOTE_STATIC) != 0 && err == NOISE_ERROR_NONE)
130  err = noise_dhstate_new_by_id(&((*state)->dh_remote_static), dh_id);
131  if ((flags & NOISE_PAT_FLAG_REMOTE_EMPEMERAL) != 0 && err == NOISE_ERROR_NONE)
132  err = noise_dhstate_new_by_id(&((*state)->dh_remote_ephemeral), dh_id);
133 
134  /* Bail out if we had an error trying to create the DHState objects */
135  if (err != NOISE_ERROR_NONE) {
136  noise_handshakestate_free(*state);
137  return err;
138  }
139 
140  /* Ready to go */
141  return NOISE_ERROR_NONE;
142 }

391 int noise_handshakestate_set_pre_shared_key
392  (NoiseHandshakeState *state, const uint8_t *key, size_t key_len)
393 {
394  uint8_t temp[NOISE_MAX_HASHLEN];
395  NoiseHashState *hash;
396 
397  /* Validate the parameters and state */
398  if (!state || !key)
399  return NOISE_ERROR_INVALID_PARAM;
400  if (state->symmetric->id.prefix_id != NOISE_PREFIX_PSK)
401  return NOISE_ERROR_NOT_APPLICABLE;
402  if (state->action != NOISE_ACTION_NONE)
403  return NOISE_ERROR_INVALID_STATE;
404  if (!(state->requirements & NOISE_REQ_PSK))
405  return NOISE_ERROR_INVALID_STATE;
406 
407  /* If we haven't hashed the prologue yet, hash an empty one now */
408  if (state->requirements & NOISE_REQ_PROLOGUE)
409  noise_handshakestate_set_prologue(state, "", 0);
410 
411  /* Mix the pre shared key into the chaining key and handshake hash */
412  hash = state->symmetric->hash;
413  noise_hashstate_hkdf
414  (hash, state->symmetric->ck, hash->hash_len, key, key_len,
415  state->symmetric->ck, hash->hash_len, temp, hash->hash_len);
416  noise_symmetricstate_mix_hash(state->symmetric, temp, hash->hash_len);
417  noise_clean(temp, sizeof(temp));
418 
419  /* We have the pre shared key now */
420  state->requirements &= ~NOISE_REQ_PSK;
421  return NOISE_ERROR_NONE;
422 }

599 int noise_handshakestate_start(NoiseHandshakeState *state)
600 {
601  /* Validate the parameter */
602  if (!state)
603  return NOISE_ERROR_INVALID_PARAM;
604  if (state->action != NOISE_ACTION_NONE)
605  return NOISE_ERROR_INVALID_STATE;
606  if (state->symmetric->id.pattern_id == NOISE_PATTERN_XX_FALLBACK &&
607  (state->requirements & NOISE_REQ_FALLBACK_PREMSG) == 0)
608  return NOISE_ERROR_NOT_APPLICABLE;
609 
610  /* Check that we have satisfied all of the pattern requirements */
611  if ((state->requirements & NOISE_REQ_LOCAL_REQUIRED) != 0 &&
612  !noise_dhstate_has_keypair(state->dh_local_static))
613  return NOISE_ERROR_LOCAL_KEY_REQUIRED;
614  if ((state->requirements & NOISE_REQ_REMOTE_REQUIRED) != 0 &&
615  !noise_dhstate_has_public_key(state->dh_remote_static))
616  return NOISE_ERROR_REMOTE_KEY_REQUIRED;
617  if ((state->requirements & NOISE_REQ_PSK) != 0)
618  return NOISE_ERROR_PSK_REQUIRED;
619 
620  /* If the prologue has not been provided yet, hash an empty one */
621  if (state->requirements & NOISE_REQ_PROLOGUE)
622  noise_handshakestate_set_prologue(state, "", 0);
623 
624  /* Mix the pre-supplied public keys into the handshake hash */
625  if (state->role == NOISE_ROLE_INITIATOR) {
626  if (state->requirements & NOISE_REQ_LOCAL_PREMSG)
627  noise_handshakestate_mix_public_key(state, state->dh_local_static);
628  if (state->requirements & NOISE_REQ_REMOTE_PREMSG)
629  noise_handshakestate_mix_public_key(state, state->dh_remote_static);
630  if (state->requirements & NOISE_REQ_FALLBACK_PREMSG)
631  noise_handshakestate_mix_public_key(state, state->dh_remote_ephemeral);
632  } else {
633  if (state->requirements & NOISE_REQ_REMOTE_PREMSG)
634  noise_handshakestate_mix_public_key(state, state->dh_remote_static);
635  if (state->requirements & NOISE_REQ_LOCAL_PREMSG)
636  noise_handshakestate_mix_public_key(state, state->dh_local_static);
637  if (state->requirements & NOISE_REQ_FALLBACK_PREMSG)
638  noise_handshakestate_mix_public_key(state, state->dh_local_ephemeral);
639  }
640 
641  /* The handshake has now officially started */
642  if (state->role == NOISE_ROLE_INITIATOR)
643  state->action = NOISE_ACTION_WRITE_MESSAGE;
644  else
645  state->action = NOISE_ACTION_READ_MESSAGE;
646  return NOISE_ERROR_NONE;
647 }

681 int noise_handshakestate_fallback(NoiseHandshakeState *state)
682 {
683  char name[NOISE_MAX_PROTOCOL_NAME];
684  size_t hash_len;
685  size_t name_len;
686  NoiseProtocolId id;
687  int err;
688 
689  /* Validate the parameter */
690  if (!state)
691  return NOISE_ERROR_INVALID_PARAM;
692  if (state->symmetric->id.pattern_id != NOISE_PATTERN_IK)
693  return NOISE_ERROR_NOT_APPLICABLE;
694 
695  /* The initiator should be waiting for a return message from the
696  responder, and the responder should have failed on the first
697  handshake message from the initiator. We also allow the
698  responder to fallback after processing the first message
699  successfully; it decides to always fall back anyway. */
700  if (state->role == NOISE_ROLE_INITIATOR) {
701  if (state->action != NOISE_ACTION_READ_MESSAGE)
702  return NOISE_ERROR_INVALID_STATE;
703  if (!noise_dhstate_has_public_key(state->dh_local_ephemeral))
704  return NOISE_ERROR_INVALID_STATE;
705  } else {
706  if (state->action != NOISE_ACTION_FAILED &&
707  state->action != NOISE_ACTION_WRITE_MESSAGE)
708  return NOISE_ERROR_INVALID_STATE;
709  if (!noise_dhstate_has_public_key(state->dh_remote_ephemeral))
710  return NOISE_ERROR_INVALID_STATE;
711  }
712 
713  /* Format a new protocol name for the "XXfallback" variant */
714  id = state->symmetric->id;
715  id.pattern_id = NOISE_PATTERN_XX_FALLBACK;
716  err = noise_protocol_id_to_name(name, sizeof(name), &id);
717  if (err != NOISE_ERROR_NONE)
718  return err;
719 
720  /* Convert the HandshakeState to the "XXfallback" pattern */
721  state->symmetric->id.pattern_id = NOISE_PATTERN_XX_FALLBACK;
722  noise_dhstate_clear_key(state->dh_remote_static);
723  if (state->role == NOISE_ROLE_INITIATOR) {
724  noise_dhstate_clear_key(state->dh_remote_ephemeral);
725  state->action = NOISE_ACTION_READ_MESSAGE;
726  state->role = NOISE_ROLE_RESPONDER;
727  } else {
728  noise_dhstate_clear_key(state->dh_local_ephemeral);
729  state->action = NOISE_ACTION_WRITE_MESSAGE;
730  state->role = NOISE_ROLE_INITIATOR;
731  }
732  state->requirements |= NOISE_REQ_FALLBACK_PREMSG;
733 
734  /* Re-initialize the chaining key "ck" and the handshake hash "h" from
735  the new protocol name. If the name is too long, hash it down first */
736  name_len = strlen(name);
737  hash_len = noise_hashstate_get_hash_length(state->symmetric->hash);
738  if (name_len <= hash_len) {
739  memcpy(state->symmetric->h, name, name_len);
740  memset(state->symmetric->h + name_len, 0, hash_len - name_len);
741  } else {
742  noise_hashstate_hash_one
743  (state->symmetric->hash, (const uint8_t *)name, name_len,
744  state->symmetric->h);
745  }
746  memcpy(state->symmetric->ck, state->symmetric->h, hash_len);
747 
748  /* Reset the encryption key within the symmetric state to empty */
749  state->symmetric->cipher->has_key = 0;
750  state->symmetric->cipher->n = 0;
751  state->symmetric->cipher->nonce_overflow = 0;
752 
753  /* Ready to go */
754  return NOISE_ERROR_NONE;
755 }

829 static int noise_handshakestate_write
830  (NoiseHandshakeState *state, const void *payload, size_t payload_size,
831  uint8_t *message, size_t *message_size, size_t max_size)
832 {
833  size_t size = 0;
834  size_t len;
835  size_t out_len;
836  size_t mac_len;
837  uint8_t token;
838  int err;
839 
840  /* Process tokens until the direction changes or the pattern ends */
841  for (;;) {
842  token = *(state->tokens);
843  if (token == NOISE_TOKEN_END) {
844  /* The pattern has finished, so the next action is "split" */
845  state->action = NOISE_ACTION_SPLIT;
846  break;
847  } else if (token == NOISE_TOKEN_FLIP_DIR) {
848  /* Changing directions, so this message is complete and
849  the next action is "read message". */
850  ++(state->tokens);
851  state->action = NOISE_ACTION_READ_MESSAGE;
852  break;
853  }
854  err = NOISE_ERROR_NONE;
855  switch (token) {
856  case NOISE_TOKEN_E:
857  /* Generate a local ephemeral keypair and add the public
858  key to the message. If we are running fixed vector tests,
859  then the ephemeral key may have already been provided. */
860  if (!state->dh_local_ephemeral)
861  return NOISE_ERROR_INVALID_STATE;
862  len = state->dh_local_ephemeral->public_key_len;
863  if (!state->dh_fixed_ephemeral) {
864  err = noise_dhstate_generate_keypair(state->dh_local_ephemeral);
865  } else {
866  /* Use the fixed ephemeral key provided by the test harness */
867  err = noise_dhstate_set_keypair
868  (state->dh_local_ephemeral,
869  state->dh_fixed_ephemeral->private_key,
870  state->dh_fixed_ephemeral->private_key_len,
871  state->dh_fixed_ephemeral->public_key,
872  state->dh_fixed_ephemeral->public_key_len);
873  }
874  if (err != NOISE_ERROR_NONE)
875  break;
876  if ((max_size - size) < len)
877  return NOISE_ERROR_INVALID_LENGTH;
878  memcpy(message + size, state->dh_local_ephemeral->public_key, len);
879  noise_symmetricstate_mix_hash
880  (state->symmetric, message + size, len);
881  size += len;
882 
883  /* If the protocol is using pre-shared keys, then also mix
884  the local ephemeral key into the chaining key */
885  if (state->symmetric->id.prefix_id == NOISE_PREFIX_PSK) {
886  err = noise_symmetricstate_mix_key
887  (state->symmetric,
888  state->dh_local_ephemeral->public_key, len);
889  }
890  break;
891  case NOISE_TOKEN_S:
892  /* Encrypt the local static public key and add it to the message */
893  if (!state->dh_local_static)
894  return NOISE_ERROR_INVALID_STATE;
895  len = state->dh_local_static->public_key_len;
896  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
897  if ((max_size - size) < (len + mac_len))
898  return NOISE_ERROR_INVALID_LENGTH;
899  memcpy(message + size, state->dh_local_static->public_key, len);
900  err = noise_symmetricstate_encrypt_and_hash
901  (state->symmetric, message + size, len, &out_len);
902  if (err != NOISE_ERROR_NONE)
903  break;
904  size += out_len;
905  break;
906  case NOISE_TOKEN_DHEE:
907  /* DH operation with local and remote ephemeral keys */
908  err = noise_handshake_mix_dh
909  (state, state->dh_local_ephemeral, state->dh_remote_ephemeral);
910  break;
911  case NOISE_TOKEN_DHES:
912  /* DH operation with local ephemeral and remote static keys */
913  err = noise_handshake_mix_dh
914  (state, state->dh_local_ephemeral, state->dh_remote_static);
915  break;
916  case NOISE_TOKEN_DHSE:
917  /* DH operation with local static and remote ephemeral keys */
918  err = noise_handshake_mix_dh
919  (state, state->dh_local_static, state->dh_remote_ephemeral);
920  break;
921  case NOISE_TOKEN_DHSS:
922  /* DH operation with local and remote static keys */
923  err = noise_handshake_mix_dh
924  (state, state->dh_local_static, state->dh_remote_static);
925  break;
926  default:
927  /* Unknown token code in the pattern. This shouldn't happen.
928  If it does, then abort immediately. */
929  err = NOISE_ERROR_INVALID_STATE;
930  break;
931  }
932  if (err != NOISE_ERROR_NONE)
933  return err;
934  ++(state->tokens);
935  }
936 
937  /* Encrypt the payload and add it to the buffer */
938  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
939  if ((max_size - size) < mac_len)
940  return NOISE_ERROR_INVALID_LENGTH;
941  if ((max_size - size - mac_len) < payload_size)
942  return NOISE_ERROR_INVALID_LENGTH;
943  if (payload_size)
944  memcpy(message + size, payload, payload_size);
945  err = noise_symmetricstate_encrypt_and_hash
946  (state->symmetric, message + size, payload_size, &out_len);
947  if (err != NOISE_ERROR_NONE)
948  return err;
949  size += out_len;
950 
951  /* Return the final size to the caller */

983 int noise_handshakestate_write_message
984  (NoiseHandshakeState *state, const void *payload, size_t payload_size,
985  uint8_t *message, size_t *message_size)
986 {
987  size_t max_size = 0;
988  int err = NOISE_ERROR_NONE;
989 
990  /* Validate the message size and extract it. We set the return size
991  to zero and clear the message buffer in case this function bails out
992  with an error later. This is in case the caller forgets to check the
993  error return value, and simply transmits the contents of "message".
994  This will prevent the caller from accidentally leaking the previous
995  contents of "message" onto the transport. */
996  if (message_size) {
997  max_size = *message_size;
998  *message_size = 0;
999  if (message)
1000  noise_clean(message, max_size);
1001  } else {
1002  err = NOISE_ERROR_INVALID_PARAM;
1003  }
1004 
1005  /* Validate the other parameters and state */
1006  if (err != NOISE_ERROR_NONE) {
1007  if (!state || !message)
1008  err = NOISE_ERROR_INVALID_PARAM;
1009  else if (!payload && payload_size != 0)
1010  err = NOISE_ERROR_INVALID_PARAM;
1011  else if (state->action != NOISE_ACTION_WRITE_MESSAGE)
1012  err = NOISE_ERROR_INVALID_STATE;
1013  }
1014 
1015  /* Perform the main write if no error so far */
1016  if (err == NOISE_ERROR_NONE) {
1017  err = noise_handshakestate_write
1018  (state, payload, payload_size, message, message_size, max_size);
1019  }
1020 
1021  /* If an error occurred, then fail the HandshakeState completely */
1022  if (err != NOISE_ERROR_NONE && state) {
1023  state->action = NOISE_ACTION_FAILED;
1024  if (message) {
1025  /* Clear the message buffer again in case we wrote some
1026  partial data to it before discovering the error */
1027  noise_clean(message, max_size);
1028  }
1029  }
1030 
1031  /* Finished */
1032  return err;
1033 }

1049 static int noise_handshakestate_read
1050  (NoiseHandshakeState *state, uint8_t *message, size_t message_size,
1051  void *payload, size_t *payload_size, size_t max_size)
1052 {
1053  size_t len;
1054  size_t out_len;
1055  size_t mac_len;
1056  uint8_t token;
1057  int err;
1058 
1059  /* Process tokens until the direction changes or the pattern ends */
1060  for (;;) {
1061  token = *(state->tokens);
1062  if (token == NOISE_TOKEN_END) {
1063  /* The pattern has finished, so the next action is "split" */
1064  state->action = NOISE_ACTION_SPLIT;
1065  break;
1066  } else if (token == NOISE_TOKEN_FLIP_DIR) {
1067  /* Changing directions, so this message is complete and
1068  the next action is "read message". */
1069  ++(state->tokens);
1070  state->action = NOISE_ACTION_WRITE_MESSAGE;
1071  break;
1072  }
1073  err = NOISE_ERROR_NONE;
1074  switch (token) {
1075  case NOISE_TOKEN_E:
1076  /* Save the remote ephemeral key and hash it */
1077  if (!state->dh_remote_ephemeral)
1078  return NOISE_ERROR_INVALID_STATE;
1079  len = state->dh_remote_ephemeral->public_key_len;
1080  if (message_size < len)
1081  return NOISE_ERROR_INVALID_LENGTH;
1082  err = noise_symmetricstate_mix_hash(state->symmetric, message, len);
1083  if (err != NOISE_ERROR_NONE)
1084  break;
1085  err = noise_dhstate_set_public_key
1086  (state->dh_remote_ephemeral, message, len);
1087  if (err != NOISE_ERROR_NONE)
1088  break;
1089  if (noise_dhstate_is_null_public_key(state->dh_remote_ephemeral)) {
1090  /* The remote ephemeral key is null, which means that it is
1091  not contributing anything to the security of the session
1092  and is in fact downgrading the security to "none at all"
1093  in some of the message patterns. Reject all such keys. */
1094  return NOISE_ERROR_INVALID_PUBLIC_KEY;
1095  }
1096  message += len;
1097  message_size -= len;
1098 
1099  /* If the protocol is using pre-shared keys, then also mix
1100  the remote ephemeral key into the chaining key */
1101  if (state->symmetric->id.prefix_id == NOISE_PREFIX_PSK) {
1102  err = noise_symmetricstate_mix_key
1103  (state->symmetric,
1104  state->dh_remote_ephemeral->public_key, len);
1105  }
1106  break;
1107  case NOISE_TOKEN_S:
1108  /* Decrypt and read the remote static key */
1109  if (!state->dh_remote_static)
1110  return NOISE_ERROR_INVALID_STATE;
1111  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
1112  len = state->dh_remote_static->public_key_len + mac_len;
1113  if (message_size < len)
1114  return NOISE_ERROR_INVALID_LENGTH;
1115  err = noise_symmetricstate_decrypt_and_hash
1116  (state->symmetric, message, len, &out_len);
1117  if (err != NOISE_ERROR_NONE)
1118  break;
1119  err = noise_dhstate_set_public_key
1120  (state->dh_remote_static, message, out_len);
1121  if (err != NOISE_ERROR_NONE)
1122  break;
1123  message += len;
1124  message_size -= len;
1125  break;
1126  case NOISE_TOKEN_DHEE:
1127  /* DH operation with local and remote ephemeral keys */
1128  err = noise_handshake_mix_dh
1129  (state, state->dh_local_ephemeral, state->dh_remote_ephemeral);
1130  break;
1131  case NOISE_TOKEN_DHES:
1132  /* DH operation with remote ephemeral and local static keys */
1133  err = noise_handshake_mix_dh
1134  (state, state->dh_local_static, state->dh_remote_ephemeral);
1135  break;
1136  case NOISE_TOKEN_DHSE:
1137  /* DH operation with remote static and local ephemeral keys */
1138  err = noise_handshake_mix_dh
1139  (state, state->dh_local_ephemeral, state->dh_remote_static);
1140  break;
1141  case NOISE_TOKEN_DHSS:
1142  /* DH operation with local and remote static keys */
1143  err = noise_handshake_mix_dh
1144  (state, state->dh_local_static, state->dh_remote_static);
1145  break;
1146  default:
1147  /* Unknown token code in the pattern. This shouldn't happen.
1148  If it does, then abort immediately. */
1149  err = NOISE_ERROR_INVALID_STATE;
1150  break;
1151  }
1152  if (err != NOISE_ERROR_NONE)
1153  return err;
1154  ++(state->tokens);
1155  }
1156 
1157  /* Decrypt the payload and return it in the payload buffer */
1158  mac_len = noise_symmetricstate_get_mac_length(state->symmetric);
1159  err = noise_symmetricstate_decrypt_and_hash
1160  (state->symmetric, message, message_size, &len);
1161  if (err != NOISE_ERROR_NONE)
1162  return err;
1163  if (len > max_size)
1164  return NOISE_ERROR_INVALID_LENGTH;
1165  if (payload)
1166  memcpy(payload, message, len);
1167 
1168  /* Return the final payload size to the caller */
1169  *payload_size = len;
1170  return NOISE_ERROR_NONE;
1171 }
