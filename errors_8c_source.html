76 static const char *noise_errstr(int err)
77 {
78  if (err == NOISE_ERROR_NONE)
79  return error_strings[0];
80  if (err < NOISE_ID('E', 1) || err >= NOISE_ID('E', num_error_strings))
81  return 0;
82  return error_strings[err - NOISE_ID('E', 0)];
