61 int noise_cipherstate_new_by_id(NoiseCipherState **state, int id)
62 {
63  /* The "state" argument must be non-NULL */
64  if (!state)
65  return NOISE_ERROR_INVALID_PARAM;
66 
67  /* Create the CipherState object for the "id" */
68  *state = 0;
69  switch (id) {
70  case NOISE_CIPHER_CHACHAPOLY:
71  *state = noise_chachapoly_new();
72  break;
73 
74  case NOISE_CIPHER_AESGCM:
75  *state = noise_aesgcm_new();
76  break;
77 
78  default:
79  return NOISE_ERROR_UNKNOWN_ID;
80  }
81 
82  /* Bail out if insufficient memory */
83  if (!(*state))
84  return NOISE_ERROR_NO_MEMORY;
85 
86  /* Ready to go */
87  return NOISE_ERROR_NONE;
88 }

263 int noise_cipherstate_encrypt_with_ad
264  (NoiseCipherState *state, const uint8_t *ad, size_t ad_len,
265  uint8_t *data, size_t in_data_len, size_t *out_data_len)
266 {
267  int err;
268 
269  /* Set the output length to zero in case of error */
270  if (!out_data_len)
271  return NOISE_ERROR_INVALID_PARAM;
272  *out_data_len = 0;
273 
274  /* Validate the parameters */
275  if (!state || (!ad && ad_len) || !data)
276  return NOISE_ERROR_INVALID_PARAM;
277 
278  /* If the key hasn't been set yet, return the plaintext as-is */
279  if (!state->has_key) {
280  if (in_data_len > NOISE_MAX_PAYLOAD_LEN)
281  return NOISE_ERROR_INVALID_LENGTH;
282  *out_data_len = in_data_len;
283  return NOISE_ERROR_NONE;
284  }
285 
286  /* Make sure that there is room for the MAC */
287  if (in_data_len > (size_t)(NOISE_MAX_PAYLOAD_LEN - state->mac_len))
288  return NOISE_ERROR_INVALID_LENGTH;
289 
290  /* If the nonce has overflowed, then further encryption is impossible */
291  if (state->nonce_overflow)
292  return NOISE_ERROR_INVALID_NONCE;
293 
294  /* Encrypt the plaintext */
295  err = (*(state->encrypt))(state, ad, ad_len, data, in_data_len);
296  if (state->n == 0xFFFFFFFFFFFFFFFFULL)
297  state->nonce_overflow = 1;
298  ++(state->n);
299  if (err != NOISE_ERROR_NONE)
300  return err;
301 
302  /* Adjust the output length for the MAC and return */
303  *out_data_len = in_data_len + state->mac_len;
304  return NOISE_ERROR_NONE;
305 }

401 {
402  /* Set the return object to NULL in case of error */
403  if (!new_state)

405  *new_state = 0;
406 
407  /* Validate the parameters */
408  if (!state || !key)
409  return NOISE_ERROR_INVALID_PARAM;
410  if (key_len != state->key_len)
411  return NOISE_ERROR_INVALID_LENGTH;
412 
413  /* Create a new CipherState object of the same type */

415  if (!(*new_state))
416  return NOISE_ERROR_NO_MEMORY;
417 
418  /* Initialize the key in the new state object */
419  noise_cipherstate_init_key(*new_state, key, key_len);
420  return NOISE_ERROR_NONE;

442 {
443  /* Bail out if the state is NULL */
444  if (!state)
445  return NOISE_ERROR_INVALID_PARAM;
446 
447  /* If the key hasn't been set yet, we cannot do this */
448  if (!state->has_key)

450 
451  /* Reject the value if the nonce would go backwards */
452  if (state->n > nonce)
