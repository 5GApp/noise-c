67 int noise_cipherstate_new_by_id(NoiseCipherState **state, int id)

93  return NOISE_ERROR_NONE;

271 int noise_cipherstate_encrypt_with_ad
272  (NoiseCipherState *state, const uint8_t *ad, size_t ad_len,
273  uint8_t *data, size_t in_data_len, size_t *out_data_len)
274 {
275  int err;
276 
277  /* Set the output length to zero in case of error */
278  if (!out_data_len)
279  return NOISE_ERROR_INVALID_PARAM;
280  *out_data_len = 0;
281 
282  /* Validate the parameters */
283  if (!state || (!ad && ad_len) || !data)
284  return NOISE_ERROR_INVALID_PARAM;
285 
286  /* If the key hasn't been set yet, return the plaintext as-is */
287  if (!state->has_key) {
288  if (in_data_len > NOISE_MAX_PAYLOAD_LEN)
289  return NOISE_ERROR_INVALID_LENGTH;
290  *out_data_len = in_data_len;
291  return NOISE_ERROR_NONE;
292  }
293 
294  /* Make sure that there is room for the MAC */
295  if (in_data_len > (size_t)(NOISE_MAX_PAYLOAD_LEN - state->mac_len))
296  return NOISE_ERROR_INVALID_LENGTH;
297 
298  /* If the nonce has overflowed, then further encryption is impossible */
299  if (state->nonce_overflow)
300  return NOISE_ERROR_INVALID_NONCE;
301 
302  /* Encrypt the plaintext */
303  err = (*(state->encrypt))(state, ad, ad_len, data, in_data_len);
304  if (state->n == 0xFFFFFFFFFFFFFFFFULL)
305  state->nonce_overflow = 1;
306  ++(state->n);
307  if (err != NOISE_ERROR_NONE)
308  return err;
309 
310  /* Adjust the output length for the MAC and return */
311  *out_data_len = in_data_len + state->mac_len;
312  return NOISE_ERROR_NONE;
313 }
