277 int noise_cipherstate_encrypt_with_ad
278  (NoiseCipherState *state, const uint8_t *ad, size_t ad_len,
279  uint8_t *data, size_t in_data_len, size_t *out_data_len)
280 {
281  int err;
282 
283  /* Set the output length to zero in case of error */
284  if (!out_data_len)
285  return NOISE_ERROR_INVALID_PARAM;
286  *out_data_len = 0;
287 
288  /* Validate the parameters */
289  if (!state || (!ad && ad_len) || !data)
290  return NOISE_ERROR_INVALID_PARAM;
291 
292  /* If the key hasn't been set yet, return the plaintext as-is */
293  if (!state->has_key) {
294  if (in_data_len > NOISE_MAX_PAYLOAD_LEN)
295  return NOISE_ERROR_INVALID_LENGTH;
296  *out_data_len = in_data_len;
297  return NOISE_ERROR_NONE;
298  }
299 
300  /* Make sure that there is room for the MAC */
301  if (in_data_len > (size_t)(NOISE_MAX_PAYLOAD_LEN - state->mac_len))
302  return NOISE_ERROR_INVALID_LENGTH;
303 
304  /* If the nonce has overflowed, then further encryption is impossible */
305  if (state->nonce_overflow)
306  return NOISE_ERROR_INVALID_NONCE;
307 
308  /* Encrypt the plaintext */
309  err = (*(state->encrypt))(state, ad, ad_len, data, in_data_len);
310  if (state->n == 0xFFFFFFFFFFFFFFFFULL)
311  state->nonce_overflow = 1;
312  ++(state->n);
313  if (err != NOISE_ERROR_NONE)
314  return err;
315 
316  /* Adjust the output length for the MAC and return */
317  *out_data_len = in_data_len + state->mac_len;
318  return NOISE_ERROR_NONE;
319 }

496 static void noise_cipherstate_rand_reseed(NoiseRandomState *rand)
497 {
498  /* Get new random data from the operating system, encrypt it
499  with the previous key/IV, and then replace the key/IV */
500  uint8_t data[40];
501  noise_rand_bytes(data, sizeof(data));
502  chacha_encrypt_bytes(&(rand->chacha), data, data, sizeof(data));
503  chacha_keysetup(&(rand->chacha), data, 256);
504  chacha_ivsetup(&(rand->chacha), data + 32, 0);
505  noise_clean(data, sizeof(data));
506  rand->left = NOISE_RAND_RESEED_COUNT;
507 }

549 int noise_cipherstate_rand_bytes
550  (NoiseCipherState *state, uint8_t *buffer, size_t len, int force_reseed)
551 {
552  /* Starting key for the random state before the first reseed.
553  This is the SHA256 initialization vector, to introduce a
554  little chaos into the starting state. */
555  static uint8_t const starting_key[32] = {
556  0x6A, 0x09, 0xE6, 0x67, 0xBB, 0x67, 0xAE, 0x85,
557  0x3C, 0x6E, 0xF3, 0x72, 0xA5, 0x4F, 0xF5, 0x3A,
558  0x51, 0x0E, 0x52, 0x7F, 0x9B, 0x05, 0x68, 0x8C,
559  0x1F, 0x83, 0xD9, 0xAB, 0x5B, 0xE0, 0xCD, 0x19
560  };
561  size_t blocks;
562 
563  /* Validate the parameters */
564  if (!state || !buffer)
565  return NOISE_ERROR_INVALID_PARAM;
566 
567  /* Create the random state for the first time if necessary */
568  if (!state->rand) {
569  state->rand = noise_new(NoiseRandomState);
570  if (!state->rand) {
571  /* Out of memory. Set the output to something just in case
572  the caller forgets to check the return value. We don't want
573  to accidentally leak the previous contents in the buffer. */
574  memset(buffer, 0xAA, len);
575  return NOISE_ERROR_NO_MEMORY;
576  }
577  chacha_keysetup(&(state->rand->chacha), starting_key, 256);
578  force_reseed = 1;
579  }
580 
581  /* Force a reseed if necessary */
582  if (force_reseed || state->rand->left < len)
583  noise_cipherstate_rand_reseed(state->rand);
584 
585  /* Generate the random data in ChaCha20 block-sized chunks */
586  memset(buffer, 0, len);
587  blocks = 0;
588  while (len > 0) {
589  size_t temp_len = len;
590  if (temp_len > 64)
591  temp_len = 64;
592  if (state->rand->left >= 64) {
593  /* One less block before we need to force a reseed */
594  state->rand->left -= 64;
595  } else {
596  /* Too much random data generated. Force a reseed now */
597  noise_cipherstate_rand_reseed(state->rand);
598  blocks = 0;
599  }
600  if (blocks++ >= NOISE_RAND_REKEY_COUNT) {
601  /* Too many blocks in the current request. Force a rekey now */
602  noise_cipherstate_rand_rekey(state->rand);
603  blocks = 0;
604  }
605  chacha_encrypt_bytes(&(state->rand->chacha), buffer, buffer, temp_len);
606  buffer += temp_len;
607  len -= temp_len;
608  }
609 
610  /* Force a rekey after every request to destroy the input that
611  was used to generate the random data for this request.
612  This prevents the state from being rolled backwards. */
613  noise_cipherstate_rand_rekey(state->rand);
614  return NOISE_ERROR_NONE;
615 }

651 int noise_cipherstate_pad
652  (NoiseCipherState *state, uint8_t *payload, size_t orig_len,
653  size_t padded_len, int padding_mode)
654 {
655  /* Validate the parameters */
656  if (!state || !payload)
657  return NOISE_ERROR_INVALID_PARAM;
658 
659  /* Nothing to do if the padded length is shorter than the original */
660  if (padded_len <= orig_len)
661  return NOISE_ERROR_NONE;
662 
663  /* Pad the payload as requested */
664  if (padding_mode == NOISE_PADDING_ZERO) {
665  memset(payload + orig_len, 0, padded_len - orig_len);
666  return NOISE_ERROR_NONE;
667  } else {
668  return noise_cipherstate_rand_bytes
669  (state, payload + orig_len, padded_len - orig_len, 0);
670  }
671 }
