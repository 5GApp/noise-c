58 static int noise_symmetricstate_new
59  (NoiseSymmetricState **state, const char *name, const NoiseProtocolId *id)
60 {
61  NoiseSymmetricState *new_state;
62  size_t name_len;
63  size_t hash_len;
64  int err;
65 
66  /* Construct a state object and initialize it */
67  new_state = noise_new(NoiseSymmetricState);
68  if (!new_state)
69  return NOISE_ERROR_NO_MEMORY;
70  new_state->id = *id;
71  err = noise_cipherstate_new_by_id(&(new_state->cipher), id->cipher_id);
72  if (err != NOISE_ERROR_NONE) {
73  noise_symmetricstate_free(new_state);
74  return err;
75  }
76  err = noise_hashstate_new_by_id(&(new_state->hash), id->hash_id);
77  if (err != NOISE_ERROR_NONE) {
78  noise_symmetricstate_free(new_state);
79  return err;
80  }
81 
82  /* Check the hash length against the maximum, just in case someone
83  adds a new hash algorithm in the future with longer output.
84  If this happens, modify NOISE_MAX_HASHLEN in internal.h accordingly */
85  hash_len = noise_hashstate_get_hash_length(new_state->hash);
86  if (hash_len > NOISE_MAX_HASHLEN) {
87  noise_symmetricstate_free(new_state);
88  return NOISE_ERROR_INVALID_LENGTH;
89  }
90 
91  /* The key length must also be less than or equal to the hash length */
92  if (noise_cipherstate_get_key_length(new_state->cipher) > hash_len) {
93  noise_symmetricstate_free(new_state);
94  return NOISE_ERROR_INVALID_LENGTH;
95  }
96 
97  /* Initialize the chaining key "ck" and the handshake hash "h" from
98  the protocol name. If the name is too long, hash it down first */
99  name_len = strlen(name);
100  if (name_len <= hash_len) {
101  memcpy(new_state->h, name, name_len);
102  memset(new_state->h + name_len, 0, hash_len - name_len);
103  } else {
104  noise_hashstate_hash_one
105  (new_state->hash, (const uint8_t *)name, name_len, new_state->h);
106  }
107  memcpy(new_state->ck, new_state->h, hash_len);
108 
109  /* Ready to go */
110  *state = new_state;
111  return NOISE_ERROR_NONE;
112 }

261 int noise_symmetricstate_mix_key
262  (NoiseSymmetricState *state, const uint8_t *input, size_t size)
263 {
264  uint8_t temp_k[NOISE_MAX_HASHLEN];
265  size_t hash_len;
266  size_t key_len;
267 
268  /* Validate the parameters */
269  if (!state || !input)
270  return NOISE_ERROR_INVALID_PARAM;
271 
272  /* If the state has been split, then we cannot do this */
273  if (!state->cipher)
274  return NOISE_ERROR_INVALID_STATE;
275 
276  /* Mix the input data in using HKDF */
277  hash_len = noise_hashstate_get_hash_length(state->hash);
278  key_len = noise_cipherstate_get_key_length(state->cipher);
279  noise_hashstate_hkdf
280  (state->hash, state->ck, hash_len, input, size,
281  state->ck, hash_len, temp_k, key_len);
282 
283  /* Change the cipher key, or set it for the first time */
284  noise_cipherstate_init_key(state->cipher, temp_k, key_len);
285  noise_clean(temp_k, sizeof(temp_k));
286  return NOISE_ERROR_NONE;
287 }

412 int noise_symmetricstate_decrypt_and_hash
413  (NoiseSymmetricState *state, uint8_t *data,
414  size_t in_data_len, size_t *out_data_len)
415 {
416  uint8_t temp[NOISE_MAX_HASHLEN];
417  size_t hash_len;
418  int err;
419 
420  /* Validate the parameters */
421  if (!out_data_len)
422  return NOISE_ERROR_INVALID_PARAM;
423  *out_data_len = 0;
424  if (!state || !data)
425  return NOISE_ERROR_INVALID_PARAM;
426 
427  /* If the state has been split, then we cannot do this */
428  if (!state->cipher)
429  return NOISE_ERROR_INVALID_STATE;
430 
431  /* Validate the input data length before we hash the data */
432  if (in_data_len > NOISE_MAX_PAYLOAD_LEN)
433  return NOISE_ERROR_INVALID_LENGTH;
434  if (noise_cipherstate_has_key(state->cipher)) {
435  if (in_data_len < noise_cipherstate_get_mac_length(state->cipher))
436  return NOISE_ERROR_INVALID_LENGTH;
437  }
438 
439  /* Feed the ciphertext into the handshake hash first. We make a
440  temporary copy of the hash. If the decryption fails below,
441  then we don't update the handshake hash with the bogus data */
442  hash_len = noise_hashstate_get_hash_length(state->hash);
443  noise_hashstate_hash_two
444  (state->hash, state->h, hash_len, data, in_data_len, temp);
445 
446  /* Decrypt the ciphertext using the underlying cipher */
447  hash_len = noise_hashstate_get_hash_length(state->hash);
448  err = noise_cipherstate_decrypt_with_ad
449  (state->cipher, state->h, hash_len, data, in_data_len, out_data_len);
450  if (err != NOISE_ERROR_NONE) {
451  noise_clean(temp, sizeof(temp));
452  return err;
453  }
454 
455  /* Update the handshake hash */
456  memcpy(state->h, temp, hash_len);
457  noise_clean(temp, sizeof(temp));
458  return NOISE_ERROR_NONE;
459 }

525 int noise_symmetricstate_split
526  (NoiseSymmetricState *state, NoiseCipherState **c1, NoiseCipherState **c2)
527 {
528  uint8_t temp_k1[NOISE_MAX_HASHLEN];
529  uint8_t temp_k2[NOISE_MAX_HASHLEN];
530  size_t hash_len;
531  size_t key_len;
532 
533  /* Validate the parameters */
534  if (!state || !c1)
535  return NOISE_ERROR_INVALID_PARAM;
536  *c1 = 0;
537  if (c2)
538  *c2 = 0;
539 
540  /* If the state has already been split, then we cannot split again */
541  if (!state->cipher)
542  return NOISE_ERROR_INVALID_STATE;
543 
544  /* Generate the two encryption keys with HKDF */
545  hash_len = noise_hashstate_get_hash_length(state->hash);
546  key_len = noise_cipherstate_get_key_length(state->cipher);
547  noise_hashstate_hkdf
548  (state->hash, state->ck, hash_len, state->ck, 0,
549  temp_k1, key_len, temp_k2, key_len);
550 
551  /* Split a copy out of the cipher and give it the second key.
552  We don't need to do this if the second CipherSuite is not required */
553  if (c2) {
554  int err = noise_cipherstate_split(state->cipher, temp_k2, key_len, c2);
555  if (err != NOISE_ERROR_NONE) {
556  noise_clean(temp_k1, sizeof(temp_k1));
557  noise_clean(temp_k2, sizeof(temp_k2));
558  return err;
559  }
560  }
561 
562  /* Re-initialize the key in the internal cipher and copy it to c1 */
563  noise_cipherstate_init_key(state->cipher, temp_k1, key_len);
564  *c1 = state->cipher;
565  state->cipher = 0;
566  noise_clean(temp_k1, sizeof(temp_k1));
567  noise_clean(temp_k2, sizeof(temp_k2));
568  return NOISE_ERROR_NONE;
569 }
