61  for (;;) {
62  int len = read(fd, bytes, size);
63  if (len == (int)size) {
64  /* We have the bytes we wanted */
65  return;
66  } else if (len >= 0) {
67  /* Short read - this shouldn't happen. Treat it as "no data" */
