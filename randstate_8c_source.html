96 int noise_randstate_new(NoiseRandState **state)
97 {
98  /* Starting key for the random state before the first reseed.
99  This is the SHA256 initialization vector, to introduce a
100  little chaos into the starting state. */
101  static uint8_t const starting_key[32] = {
102  0x6A, 0x09, 0xE6, 0x67, 0xBB, 0x67, 0xAE, 0x85,
103  0x3C, 0x6E, 0xF3, 0x72, 0xA5, 0x4F, 0xF5, 0x3A,
104  0x51, 0x0E, 0x52, 0x7F, 0x9B, 0x05, 0x68, 0x8C,
105  0x1F, 0x83, 0xD9, 0xAB, 0x5B, 0xE0, 0xCD, 0x19
106  };
107 
108  /* Validate the parameter */
109  if (!state)
110  return NOISE_ERROR_INVALID_PARAM;
111 
112  /* Create the random number generator state */
113  *state = noise_new(NoiseRandState);
114  if (!(*state))
115  return NOISE_ERROR_NO_MEMORY;
116 
117  /* Initialize the random number generator */
118  chacha_keysetup(&((*state)->chacha), starting_key, 256);
119  noise_randstate_reseed((*state));
120  return NOISE_ERROR_NONE;
121 }

163 int noise_randstate_reseed(NoiseRandState *state)
164 {
165  uint8_t data[40];
166 
167  /* Validate the parameter */
168  if (!state)
169  return NOISE_ERROR_INVALID_PARAM;
170 
171  /* Get new random data from the operating system, encrypt it
172  with the previous key/IV, and then replace the key/IV */
173  noise_rand_bytes(data, sizeof(data));
174  chacha_encrypt_bytes(&(state->chacha), data, data, sizeof(data));
175  chacha_keysetup(&(state->chacha), data, 256);
176  chacha_ivsetup(&(state->chacha), data + 32, 0);
177  state->left = NOISE_RAND_RESEED_COUNT;
178 
179  /* And force a rekey as well for good measure */
180  memset(data, 0, sizeof(data));
181  chacha_encrypt_bytes(&(state->chacha), data, data, sizeof(data));
182  chacha_keysetup(&(state->chacha), data, 256);
183  chacha_ivsetup(&(state->chacha), data + 32, 0);
184  noise_clean(data, sizeof(data));
185 
186  /* Ready to go */
187  return NOISE_ERROR_NONE;
188 }

222 int noise_randstate_generate
223  (NoiseRandState *state, uint8_t *buffer, size_t len)
224 {
225  size_t blocks;
226  size_t temp_len;
227 
228  /* Validate the parameters. We make sure to set the contents of
229  the buffer to zero before proceeding just in case this function
230  bails out before it can actually generate data. We don't want
231  to accidentally leak the previous contents in the buffer. */
232  if (!buffer)
233  return NOISE_ERROR_INVALID_PARAM;
234  memset(buffer, 0, len);
235  if (!state)
236  return NOISE_ERROR_INVALID_PARAM;
237 
238  /* Force a reseed if necessary */
239  if (state->left < len)
240  noise_randstate_reseed(state);
241 
242  /* Generate the random data in ChaCha20 block-sized chunks */
243  blocks = 0;
244  while (len > 0) {
245  temp_len = len;
246  if (temp_len > 64)
247  temp_len = 64;
248  if (state->left >= 64) {
249  /* One less block before we need to force a reseed */
250  state->left -= 64;
251  } else {
252  /* Too much random data generated. Force a reseed now */
253  noise_randstate_reseed(state);
254  blocks = 0;
255  }
256  if (blocks++ >= NOISE_RAND_REKEY_COUNT) {
257  /* Too many blocks in the current request. Force a rekey now */
258  noise_randstate_rekey(state);
259  blocks = 0;
260  }
261  chacha_encrypt_bytes(&(state->chacha), buffer, buffer, temp_len);
262  buffer += temp_len;
263  len -= temp_len;
264  }
265 
266  /* Force a rekey after every request to destroy the input that
267  was used to generate the random data for this request.
268  This prevents the state from being rolled backwards. */
269  noise_randstate_rekey(state);
270  return NOISE_ERROR_NONE;
271 }

303 int noise_randstate_pad
304  (NoiseRandState *state, uint8_t *payload, size_t orig_len,
305  size_t padded_len, int padding_mode)
306 {
307  /* Validate the parameters */
308  if (!payload)
309  return NOISE_ERROR_INVALID_PARAM;
310  if (!state) {
311  /* Zero the padding region just in case so that we don't accidentally
312  leak the previous contents of the payload's padding region. */
313  if (padded_len > orig_len)
314  memset(payload + orig_len, 0, padded_len - orig_len);
315  return NOISE_ERROR_INVALID_PARAM;
316  }
317 
318  /* Nothing to do if the new length is shorter than the original */
319  if (padded_len <= orig_len)
320  return NOISE_ERROR_NONE;
321 
322  /* Pad the payload as requested */
323  if (padding_mode == NOISE_PADDING_ZERO) {
324  memset(payload + orig_len, 0, padded_len - orig_len);
325  return NOISE_ERROR_NONE;
326  } else {
327  return noise_randstate_generate
328  (state, payload + orig_len, padded_len - orig_len);
329  }
330 }
