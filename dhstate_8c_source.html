67 int noise_dhstate_new_by_id(NoiseDHState **state, int id)
68 {
69  /* The "state" argument must be non-NULL */
70  if (!state)
71  return NOISE_ERROR_INVALID_PARAM;
72 
73  /* Create the DHState object for the "id" */
74  *state = 0;
75  switch (id) {
76  case NOISE_DH_CURVE25519:
77  *state = noise_curve25519_new();
78  break;
79 
80  case NOISE_DH_CURVE448:
81  *state = noise_curve448_new();
82  break;
83 
84  default:
85  return NOISE_ERROR_UNKNOWN_ID;
86  }
87 
88  /* Bail out if insufficient memory */
89  if (!(*state))
90  return NOISE_ERROR_NO_MEMORY;
91 
92  /* Ready to go */
93  return NOISE_ERROR_NONE;
94 }

447 int noise_dhstate_set_public_key
448  (NoiseDHState *state, const uint8_t *public_key, size_t public_key_len)
449 {
450  int is_null, err;
451 
452  /* Validate the parameters */
453  if (!state || !public_key)
454  return NOISE_ERROR_INVALID_PARAM;
455  if (public_key_len != state->public_key_len)
456  return NOISE_ERROR_INVALID_LENGTH;
457 
458  /* Validate the public key with the back end and then ignore the
459  result if the public key is the special null value */
460  is_null = noise_is_zero(public_key, public_key_len);
461  err = (*(state->validate_public_key))(state, public_key);
462  err &= (is_null - 1);
463  if (err != NOISE_ERROR_NONE)
464  return err;
465 
466  /* Copy the public key into place and clear the private key */
467  memcpy(state->public_key, public_key, public_key_len);
468  memset(state->private_key, 0, state->private_key_len);
469  state->key_type = NOISE_KEY_TYPE_PUBLIC;
470  return NOISE_ERROR_NONE;
471 }

596 int noise_dhstate_calculate
597  (const NoiseDHState *private_key_state,
598  const NoiseDHState *public_key_state,
599  uint8_t *shared_key, size_t shared_key_len)
600 {
601  int is_null, err;
602 
603  /* Validate the parameters */
604  if (!private_key_state || !public_key_state || !shared_key)
605  return NOISE_ERROR_INVALID_PARAM;
606  if (private_key_state->dh_id != public_key_state->dh_id)
607  return NOISE_ERROR_INVALID_PARAM;
608  if (shared_key_len != private_key_state->shared_key_len)
609  return NOISE_ERROR_INVALID_LENGTH;
610  if (private_key_state->key_type != NOISE_KEY_TYPE_KEYPAIR)
611  return NOISE_ERROR_INVALID_PRIVATE_KEY;
612 
613  /* If the public key is null, then the output must be null too.
614  We check for null now, but still perform the normal evaluation.
615  At the end we will null out the result in constant time */
616  is_null = noise_is_zero
617  (public_key_state->public_key, public_key_state->public_key_len);
618 
619  /* Perform the calculation */
620  err = (*(private_key_state->calculate))
621  (private_key_state, public_key_state, shared_key);
622 
623  /* If the public key was null, then we need to set the shared key
624  to null and replace any error we got from the back end with "none" */
625  noise_cmove_zero(shared_key, shared_key_len, is_null);
626  err &= (is_null - 1);
627  return err;
628 }
