59 int noise_dhstate_new_by_id(NoiseDHState **state, int id)
60 {
61  /* The "state" argument must be non-NULL */
62  if (!state)
63  return NOISE_ERROR_INVALID_PARAM;
64 
65  /* Create the DHState object for the "id" */
66  *state = 0;
67  switch (id) {
68  case NOISE_DH_CURVE25519:
69  *state = noise_curve25519_new();
70  break;
71 
72  case NOISE_DH_CURVE448:
73  *state = noise_curve448_new();
74  break;
75 
76  default:
77  return NOISE_ERROR_UNKNOWN_ID;
78  }
79 
80  /* Bail out if insufficient memory */
81  if (!(*state))
82  return NOISE_ERROR_NO_MEMORY;
83 
84  /* Ready to go */
85  return NOISE_ERROR_NONE;
86 }

218 int noise_dhstate_get_null_public_key
219  (const NoiseDHState *state, uint8_t *key, size_t len)
220 {
221  /* Validate the parameters */
222  if (!state || !key)
223  return NOISE_ERROR_INVALID_PARAM;
224 
225  /* Check the key length. We allow both the public and shared
226  key lengths because we also need to check the output of
227  noise_dhstate_calculate() for null values */
228  if (len != state->public_key_len && len != state->shared_key_len)
229  return NOISE_ERROR_INVALID_LENGTH;
230 
231  /* For now we assume that all DH algorithms use all-zeroes
232  as their null value. If this changes in the future, then
233  we will add a new back end function to NoiseDHState_s */
234  memset(key, 0, len);
235  return NOISE_ERROR_NONE;
236 }

325 int noise_dhstate_calculate
326  (const NoiseDHState *state, uint8_t *shared_key, size_t shared_key_len,
327  const uint8_t *private_key, size_t private_key_len,
328  const uint8_t *public_key, size_t public_key_len)
329 {
330  int is_null, err;
331 
332  /* Validate the parameters */
333  if (!state || !shared_key || !private_key || !public_key)
334  return NOISE_ERROR_INVALID_PARAM;
335  if (shared_key_len != state->shared_key_len)
336  return NOISE_ERROR_INVALID_LENGTH;
337  if (private_key_len != state->private_key_len)
338  return NOISE_ERROR_INVALID_LENGTH;
339  if (public_key_len != state->public_key_len)
340  return NOISE_ERROR_INVALID_LENGTH;
341 
342  /* If the public key is null, then the output must be null too.
343  We check for null now, but still perform the normal evaluation.
344  At the end we will null out the result in constant time */
345  is_null = noise_is_zero(public_key, public_key_len);
346 
347  /* Perform the calculation */
348  err = (*(state->calculate))(state, shared_key, private_key, public_key);
349 
350  /* If the public key was null, then we need to set the shared key
351  to null and replace any error we got from the back end with "none" */
352  noise_cmove_zero(shared_key, shared_key_len, is_null);
353  err &= (is_null - 1);
354  return err;
355 }
