67 int noise_dhstate_new_by_id(NoiseDHState **state, int id)
68 {
69  /* The "state" argument must be non-NULL */
70  if (!state)
71  return NOISE_ERROR_INVALID_PARAM;
72 
73  /* Create the DHState object for the "id" */
74  *state = 0;
75  switch (id) {
76  case NOISE_DH_CURVE25519:
77  *state = noise_curve25519_new();
78  break;
79 
80  case NOISE_DH_CURVE448:
81  *state = noise_curve448_new();
82  break;
83 
84  case NOISE_DH_NEWHOPE:
85  *state = noise_newhope_new();
86  break;
87 
88  default:
89  return NOISE_ERROR_UNKNOWN_ID;
90  }
91 
92  /* Bail out if insufficient memory */
93  if (!(*state))
94  return NOISE_ERROR_NO_MEMORY;
95 
96  /* Ready to go */
97  return NOISE_ERROR_NONE;
98 }

516 int noise_dhstate_set_public_key
517  (NoiseDHState *state, const uint8_t *public_key, size_t public_key_len)
518 {
519  int is_null, err;
520 
521  /* Validate the parameters */
522  if (!state || !public_key)
523  return NOISE_ERROR_INVALID_PARAM;
524  if (public_key_len != state->public_key_len)
525  return NOISE_ERROR_INVALID_LENGTH;
526 
527  /* Validate the public key with the back end and then ignore the
528  result if the public key is the special null value */
529  is_null = state->nulls_allowed & noise_is_zero(public_key, public_key_len);
530  err = (*(state->validate_public_key))(state, public_key);
531  err &= (is_null - 1);
532  if (err != NOISE_ERROR_NONE)
533  return err;
534 
535  /* Copy the public key into place and clear the private key */
536  memcpy(state->public_key, public_key, public_key_len);
537  memset(state->private_key, 0, state->private_key_len);
538  state->key_type = NOISE_KEY_TYPE_PUBLIC;
539  return NOISE_ERROR_NONE;
540 }

649 static void noise_cmove_zero(uint8_t *data, size_t len, int condition)
650 {
651  /* Turn the condition into an all-zeroes or all-ones mask.
652  If the condition is set, then we want all-zeroes in the mask.
653  If the condition is not set, then we want all-ones in the mask. */
654  uint8_t mask = ~((uint8_t)(-condition));
655 
656  /* AND the contents of the data buffer with the mask */
657  while (len > 0) {
658  *data++ &= mask;
659  --len;
660  }
661 }

689 int noise_dhstate_calculate
690  (const NoiseDHState *private_key_state,
691  const NoiseDHState *public_key_state,
692  uint8_t *shared_key, size_t shared_key_len)
693 {
694  int is_null, err;
695 
696  /* Validate the parameters */
697  if (!private_key_state || !public_key_state || !shared_key)
698  return NOISE_ERROR_INVALID_PARAM;
699  if (private_key_state->dh_id != public_key_state->dh_id)
700  return NOISE_ERROR_INVALID_PARAM;
701  if (shared_key_len != private_key_state->shared_key_len)
702  return NOISE_ERROR_INVALID_LENGTH;
703  if (private_key_state->key_type != NOISE_KEY_TYPE_KEYPAIR)
704  return NOISE_ERROR_INVALID_PRIVATE_KEY;
705 
706  /* If the public key is null, then the output must be null too.
707  We check for null now, but still perform the normal evaluation.
708  At the end we will null out the result in constant time */
709  is_null = public_key_state->nulls_allowed & noise_is_zero
710  (public_key_state->public_key, public_key_state->public_key_len);
711 
712  /* Perform the calculation */
713  err = (*(private_key_state->calculate))
714  (private_key_state, public_key_state, shared_key);
715 
716  /* If the public key was null, then we need to set the shared key
717  to null and replace any error we got from the back end with "none" */
718  noise_cmove_zero(shared_key, shared_key_len, is_null);
719  err &= (is_null - 1);
720  return err;
721 }
